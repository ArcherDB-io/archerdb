            uint64_t* ids = malloc(result->event_count * sizeof(uint64_t));

    //! Comprehensive integration tests using Phase 11 fixtures
    //!
    //! These tests load JSON fixtures and verify each SDK operation
    //! produces expected results. Run with:
    //!   ARCHERDB_INTEGRATION=1 zig build test:integration

    const std = @import("std");
    const Client = @import("../../client.zig").Client;
    const types = @import("../../types.zig");

    const FixtureDir = "../../../../test_infrastructure/fixtures/v1/";

    fn getServerUrl() []const u8 {
        return std.posix.getenv("ARCHERDB_URL") orelse "http://localhost:3001";
    }

    fn loadFixture(allocator: std.mem.Allocator, operation: []const u8) !std.json.Parsed(std.json.Value) {
        const path = FixtureDir ++ operation ++ ".json";
        const file = try std.fs.cwd().openFile(path, .{});
        defer file.close();

        const content = try file.readToEndAlloc(allocator, 1024 * 1024);
        defer allocator.free(content);

        return std.json.parseFromSlice(std.json.Value, allocator, content, .{});
    }

    fn cleanDatabase(client: *Client, allocator: std.mem.Allocator) !void {
        const filter = types.QueryLatestFilter{ .limit = 10000 };
        var result = try client.queryLatest(allocator, filter);
        defer result.deinit();

        if (result.events.items.len > 0) {
            var ids = try allocator.alloc(u128, result.events.items.len);
            defer allocator.free(ids);

            for (result.events.items, 0..) |event, i| {
                ids[i] = event.entity_id;
            }
            _ = try client.deleteEntities(allocator, ids);
        }
    }

    // === Insert Operation Tests (opcode 146) ===
    test "fixture: insert operations" {
        var client = Client.init(std.testing.allocator, getServerUrl()) catch {
            std.debug.print("Skipping: server not available\n", .{});
            return;
        };
        defer client.deinit();

        _ = client.ping() catch {
            std.debug.print("Skipping: server not responding\n", .{});
            return;
        };

        var fixture = try loadFixture(std.testing.allocator, "insert");
        defer fixture.deinit();

        const cases = fixture.value.object.get("cases").?.array;
        for (cases.items) |case| {
            const name = case.object.get("name").?.string;
            std.debug.print("  Running: {s}\n", .{name});

            try cleanDatabase(&client, std.testing.allocator);

            // Parse and execute test case
            const input = case.object.get("input").?;
            const events_json = input.object.get("events").?.array;

            var events = try std.testing.allocator.alloc(types.GeoEvent, events_json.items.len);
            defer std.testing.allocator.free(events);

            for (events_json.items, 0..) |event_json, i| {
                events[i] = .{
                    .entity_id = @intCast(event_json.object.get("entity_id").?.integer),
                    .lat_nano = types.degreesToNano(event_json.object.get("latitude").?.float),
                    .lon_nano = types.degreesToNano(event_json.object.get("longitude").?.float),
                    .group_id = 0,
                };
            }

            var result = try client.insertEvents(std.testing.allocator, events);
            defer result.deinit();

            // Verify against expected_output
            const expected = case.object.get("expected_output").?;
            const expected_code = expected.object.get("result_code").?.integer;
            try std.testing.expectEqual(@as(i64, 0), expected_code);
        }
    }
