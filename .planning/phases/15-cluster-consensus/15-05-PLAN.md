Key structs:
```zig
pub const ReplicaHealth = struct {
    id: u128,
    is_healthy: bool,
    last_heartbeat: i64,
    replication_lag_ops: u64,
    // Health is determined by heartbeat freshness from timeout_profiles
};

pub const RouteDecision = struct {
    target: enum { leader, replica },
    replica_id: ?u128,  // Set if target == .replica
    reason: RouteReason,
};

pub const RouteReason = enum {
    write_operation,       // Always to leader
    read_to_replica,       // Healthy replica available
    read_failover_leader,  // No healthy replicas
    explicit_leader,       // Client requested leader
};

pub const ReadReplicaRouter = struct {
    leader_id: u128,
    replicas: []ReplicaHealth,
    round_robin_index: std.atomic.Value(usize),
    metrics: *ClusterMetrics,
    health_check_interval_ms: u64,  // From timeout_profiles

    pub fn route(self: *Self, query_type: QueryType) RouteDecision {
        // Automatic classification
        if (!query_type.isReadOnly()) {
            self.metrics.recordRoute(.write_operation);
            return .{ .target = .leader, .replica_id = null, .reason = .write_operation };
        }

        // Select healthy replica (round-robin)
        if (self.selectHealthyReplica()) |replica_id| {
            self.metrics.recordRoute(.read_to_replica);
            return .{ .target = .replica, .replica_id = replica_id, .reason = .read_to_replica };
        }

        // Fail to leader if all replicas unhealthy
        self.metrics.recordRoute(.read_failover_leader);
        return .{ .target = .leader, .replica_id = null, .reason = .read_failover_leader };
    }

    fn selectHealthyReplica(self: *Self) ?u128 { ... }
    pub fn updateReplicaHealth(self: *Self, replica_id: u128, healthy: bool) void { ... }
    pub fn updateReplicationLag(self: *Self, replica_id: u128, lag_ops: u64) void { ... }
};

pub const QueryType = struct {
    has_mutation: bool,     // INSERT, UPDATE, DELETE, etc.
    has_transaction: bool,  // BEGIN, COMMIT, etc.

    pub fn isReadOnly(self: *const Self) bool {
        return !self.has_mutation and !self.has_transaction;
    }
};
```
