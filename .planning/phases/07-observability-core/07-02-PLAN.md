Create new directory `src/archerdb/observability/` and file `correlation.zig`:

1. Define CorrelationContext struct:
```zig
pub const CorrelationContext = struct {
    /// W3C trace-id (16 bytes, 32 hex chars)
    trace_id: [16]u8,
    /// W3C parent-id / span-id (8 bytes, 16 hex chars)
    span_id: [8]u8,
    /// Trace flags (sampled, etc.)
    flags: u8,
    /// Internal request ID for correlation
    request_id: u128,
    /// Replica that received the request
    replica_id: u8,

    /// Parse from W3C traceparent header
    /// Format: version-trace_id-parent_id-flags (e.g., "00-0af7651916cd43dd8448eb211c80319c-b7ad6b7169203331-01")
    pub fn fromTraceparent(header: []const u8) ?CorrelationContext

    /// Parse from B3 headers (X-B3-TraceId, X-B3-SpanId, X-B3-Sampled)
    pub fn fromB3Headers(trace_id: ?[]const u8, span_id: ?[]const u8, sampled: ?[]const u8) ?CorrelationContext

    /// Generate new root context (when no incoming trace)
    pub fn newRoot(replica_id: u8) CorrelationContext

    /// Generate child span context
    pub fn newChild(self: *const CorrelationContext) CorrelationContext

    /// Format as W3C traceparent header value
    pub fn toTraceparent(self: *const CorrelationContext, buf: []u8) []const u8

    /// Format trace_id as lowercase hex (for JSON/logs)
    pub fn traceIdHex(self: *const CorrelationContext) [32]u8

    /// Format span_id as lowercase hex
    pub fn spanIdHex(self: *const CorrelationContext) [16]u8

    /// Check if sampled flag is set
    pub fn isSampled(self: *const CorrelationContext) bool
};
```

2. Implement parsing logic:
   - W3C traceparent: split on '-', validate version=00, parse hex
   - B3: accept 64-bit or 128-bit trace IDs, pad 64-bit to 128-bit
   - Generate random trace_id/span_id using std.crypto.random

3. Add thread-local storage for current context:
```zig
threadlocal var current_context: ?*const CorrelationContext = null;

pub fn setCurrent(ctx: ?*const CorrelationContext) void
pub fn getCurrent() ?*const CorrelationContext
```

This follows W3C Trace Context spec: https://www.w3.org/TR/trace-context/

<task type="auto">
  <name>Task 2: Create OTLP trace exporter</name>
  <files>src/archerdb/observability/trace_export.zig</files>
  <action>
Create `trace_export.zig` with OTLP JSON exporter:

1. Define Span struct matching OTLP schema:
```zig
pub const Span = struct {
    trace_id: [16]u8,
    span_id: [8]u8,
    parent_span_id: ?[8]u8,
    name: []const u8,
    kind: SpanKind,
    start_time_ns: i128,
    end_time_ns: i128,
    attributes: []const Attribute,
    events: []const SpanEvent,
    status: SpanStatus,

    pub const SpanKind = enum { internal, server, client };
    pub const SpanStatus = enum { unset, ok, error };
    pub const Attribute = struct { key: []const u8, value: AttributeValue };
    pub const AttributeValue = union(enum) { string: []const u8, int: i64, float: f64, bool: bool };
    pub const SpanEvent = struct { name: []const u8, time_ns: i128, attributes: []const Attribute };
};
```

2. Implement OtlpTraceExporter:
```zig
pub const OtlpTraceExporter = struct {
    endpoint: []const u8,  // e.g., "http://localhost:4318/v1/traces"
    buffer: std.ArrayList(Span),
    allocator: std.mem.Allocator,
    max_batch_size: usize = 100,
    flush_interval_ns: i128 = 5_000_000_000,  // 5 seconds
    last_flush_ns: i128 = 0,

    pub fn init(allocator: std.mem.Allocator, endpoint: []const u8) OtlpTraceExporter
    pub fn deinit(self: *OtlpTraceExporter) void

    /// Record a span (non-blocking, buffers internally)
    pub fn recordSpan(self: *OtlpTraceExporter, span: Span) void

    /// Flush buffered spans to collector (async via thread)
    pub fn flush(self: *OtlpTraceExporter) void

    /// Format spans as OTLP JSON
    fn formatOtlpJson(self: *OtlpTraceExporter, writer: anytype) !void
};
```

3. OTLP JSON format (per https://opentelemetry.io/docs/specs/otlp/):
```json
{
  "resourceSpans": [{
    "resource": {
      "attributes": [
        {"key": "service.name", "value": {"stringValue": "archerdb"}},
        {"key": "service.version", "value": {"stringValue": "0.0.1"}}
      ]
    },
    "scopeSpans": [{
      "scope": {"name": "archerdb"},
      "spans": [/* span objects */]
    }]
  }]
}
```

4. Use HTTP POST to send (use std.http.Client or manual socket):
   - Content-Type: application/json
   - Timeout: 5 seconds
   - Drop spans on failure (no retry, per RESEARCH.md anti-pattern)

Buffer spans in-process, export asynchronously. Never block request path.
