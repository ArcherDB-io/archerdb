```zig
fn performHealthChecks(allocator: std.mem.Allocator) ![]ComponentCheck {
    var checks = std.ArrayList(ComponentCheck).init(allocator);

    // Replica status
    try checks.append(.{
        .name = "replica",
        .status = if (replica_state.isReady()) .pass else .fail,
        .message = replica_state.reason(),
        .duration_ms = null,
    });

    // Memory check (warn if >90% of limit)
    const mem_used = metrics.Registry.memory_allocated_bytes.get();
    const mem_limit = getMemoryLimit(); // from config
    const mem_pct = @divFloor(mem_used * 100, mem_limit);
    try checks.append(.{
        .name = "memory",
        .status = if (mem_pct > 95) .fail else if (mem_pct > 90) .warn else .pass,
        .message = if (mem_pct > 90) "high memory usage" else null,
        .duration_ms = null,
    });

    // Storage check (can we write?)
    // Note: actual disk check might be too slow for health endpoint
    // Instead, check recent error metrics
    const storage_errors = metrics.Registry.write_errors_total.get();
    try checks.append(.{
        .name = "storage",
        .status = if (storage_errors > last_storage_errors + 10) .fail else .pass,
        .message = if (storage_errors > 0) "storage errors detected" else null,
        .duration_ms = null,
    });

    // S3/replication check (if configured)
    const repl_lag = metrics.Registry.replication_lag_ns.load(.monotonic);
    if (repl_lag > 0) {
        const lag_seconds = @divFloor(repl_lag, 1_000_000_000);
        try checks.append(.{
            .name = "replication",
            .status = if (lag_seconds > 60) .fail else if (lag_seconds > 30) .warn else .pass,
            .message = if (lag_seconds > 30) "high replication lag" else null,
            .duration_ms = null,
        });
    }

    return checks.toOwnedSlice();
}
```

```zig
fn handleHealthReady(client_fd: posix.socket_t) !void {
    // Must be initialized AND ready
    if (!server_initialized) {
        const body = "{\"status\":\"initializing\",\"reason\":\"server starting\"}";
        try sendResponse(client_fd, .service_unavailable, "application/json", body);
        return;
    }

    const state = replica_state;
    const uptime = getUptimeSeconds();

    if (state.isReady()) {
        var body_buf: [512]u8 = undefined;
        const body = std.fmt.bufPrint(&body_buf,
            "{{\"status\":\"ok\",\"uptime_seconds\":{d},\"version\":\"{s}\",\"commit\":\"{s}\"}}",
            .{ uptime, build_version, build_commit }
        ) catch "{\"status\":\"ok\"}";
        try sendResponse(client_fd, .ok, "application/json", body);
    } else {
        // Return reason
        var body_buf: [512]u8 = undefined;
        const body = std.fmt.bufPrint(&body_buf,
            "{{\"status\":\"unavailable\",\"reason\":\"{s}\",\"uptime_seconds\":{d}}}",
            .{ state.reason(), uptime }
        ) catch "{\"status\":\"unavailable\"}";
        try sendResponse(client_fd, .service_unavailable, "application/json", body);
    }
}
```

```zig
const HttpStatus = enum {
    ok,                    // 200
    bad_request,           // 400
    unauthorized,          // 401
    not_found,             // 404
    too_many_requests,     // 429 (degraded)
    service_unavailable,   // 503

    fn code(self: HttpStatus) []const u8 {
        return switch (self) {
            .ok => "200 OK",
            .bad_request => "400 Bad Request",
            .unauthorized => "401 Unauthorized",
            .not_found => "404 Not Found",
            .too_many_requests => "429 Too Many Requests",
            .service_unavailable => "503 Service Unavailable",
        };
    }
};
```

5. Helper function for uptime:
```zig
fn getUptimeSeconds() u64 {
    if (server_start_time_ns == 0) return 0;
    const now = std.time.nanoTimestamp();
    const elapsed_ns = now - server_start_time_ns;
    return @intCast(@divFloor(elapsed_ns, 1_000_000_000));
}

pub fn setStartTime() void {
    server_start_time_ns = std.time.nanoTimestamp();
}
```
