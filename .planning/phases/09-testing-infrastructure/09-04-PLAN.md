Update each SDK job (Python, Node.js, Java, Go):
```yaml
test-sdk-python:
  needs: test  # Ensures binary is built
  runs-on: ubuntu-latest
  name: SDK Test (Python)
  steps:
    - uses: actions/checkout@v4
    - name: Download Zig
      run: ./zig/download.sh
    - name: Build ArcherDB
      run: ./zig/zig build -j4
    - name: Start ArcherDB
      run: |
        ./zig-out/bin/archerdb start --dev &
        sleep 5  # Wait for startup
        curl --retry 10 --retry-delay 1 http://localhost:3000/health/ready
    - uses: actions/setup-python@v5
      with:
        python-version: '3.11'
    - name: Install and run tests
      run: |
        cd src/clients/python
        pip install pytest
        python -m pytest tests/ -v
      env:
        ARCHERDB_HOST: localhost
        ARCHERDB_PORT: 3000
```
