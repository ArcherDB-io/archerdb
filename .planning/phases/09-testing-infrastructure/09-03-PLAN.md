```bash
# scripts/stress-test.sh - Stress testing for TEST-04
set -euo pipefail

DURATION="${1:-7m}"  # Default 7 minutes for CI

# Convert duration to seconds
parse_duration() {
  local dur="$1"
  case "$dur" in
    *m) echo $((${dur%m} * 60)) ;;
    *h) echo $((${dur%h} * 3600)) ;;
    *) echo "$dur" ;;
  esac
}

DURATION_SECS=$(parse_duration "$DURATION")

# Run benchmark in loop until duration expires
# Monitor memory and throughput stability
```
