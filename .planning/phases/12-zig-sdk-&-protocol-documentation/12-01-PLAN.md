    **http.zig:**
    - HttpClient struct wrapping std.http.Client:
      ```zig
      pub const HttpClient = struct {
          inner: std.http.Client,
          allocator: std.mem.Allocator,

          pub fn init(allocator: std.mem.Allocator) HttpClient { ... }
          pub fn deinit(self: *HttpClient) void { ... }
          pub fn doPost(self: *HttpClient, allocator: std.mem.Allocator, url: []const u8, body: []const u8) ![]u8 { ... }
          pub fn doGet(self: *HttpClient, allocator: std.mem.Allocator, url: []const u8) ![]u8 { ... }
      };
      ```
    - doPost(allocator, path, body) -> []u8 - POST with JSON body
    - doGet(allocator, path) -> []u8 - GET for status/topology
    - Set Content-Type: application/json header
    - Handle connection errors -> ClientError.ConnectionFailed
    - Handle timeouts -> ClientError.OperationTimeout
    - Read response body into allocator-owned slice

    **client.zig:**
    Per CONTEXT.md API design decisions. NOTE: http_client field uses HttpClient from http.zig, NOT std.http.Client directly:
    ```zig
    const http = @import("http.zig");
    const types = @import("types.zig");
    const errors = @import("errors.zig");

    pub const Client = struct {
        allocator: std.mem.Allocator,
        http_client: http.HttpClient,  // Uses HttpClient wrapper, not std.http.Client
        base_url: []const u8,

        pub fn init(allocator: std.mem.Allocator, base_url: []const u8) !Client
        pub fn deinit(self: *Client) void

        // All 14 operations - pass allocator per CONTEXT.md
        // 1. Insert
        pub fn insertEvents(self: *Client, allocator: Allocator, events: []const GeoEvent) ClientError![]InsertResult
        // 2. Upsert
        pub fn upsertEvents(self: *Client, allocator: Allocator, events: []const GeoEvent) ClientError![]InsertResult
        // 3. Delete
        pub fn deleteEntities(self: *Client, allocator: Allocator, entity_ids: []const u128) ClientError!DeleteResult
        // 4. Query by UUID
        pub fn getLatestByUUID(self: *Client, allocator: Allocator, entity_id: u128) ClientError!?GeoEvent
        // 5. Query UUID Batch
        pub fn queryUUIDBatch(self: *Client, allocator: Allocator, entity_ids: []const u128) ClientError!QueryUUIDBatchResult
        // 6. Query Radius
        pub fn queryRadius(self: *Client, allocator: Allocator, filter: QueryRadiusFilter) ClientError!QueryResult
        // 7. Query Polygon
        pub fn queryPolygon(self: *Client, allocator: Allocator, filter: QueryPolygonFilter) ClientError!QueryResult
        // 8. Query Latest
        pub fn queryLatest(self: *Client, allocator: Allocator, filter: QueryLatestFilter) ClientError!QueryResult
        // 9. Ping
        pub fn ping(self: *Client) ClientError!bool
        // 10. Status
        pub fn getStatus(self: *Client, allocator: Allocator) ClientError!StatusResponse
        // 11. Topology
        pub fn getTopology(self: *Client, allocator: Allocator) ClientError!TopologyResponse
        // 12. Set TTL
        pub fn setTTL(self: *Client, entity_id: u128, ttl_seconds: u32) ClientError!TtlSetResponse
        // 13. Extend TTL
        pub fn extendTTL(self: *Client, entity_id: u128, extend_by_seconds: u32) ClientError!TtlExtendResponse
        // 14. Clear TTL
        pub fn clearTTL(self: *Client, entity_id: u128) ClientError!TtlClearResponse
    };
    ```

    **build.zig:**
    Create standard Zig build configuration:
    ```zig
    const std = @import("std");

    pub fn build(b: *std.Build) void {
        const target = b.standardTargetOptions(.{});
        const optimize = b.standardOptimizeOption(.{});

        // Library
        const lib = b.addStaticLibrary(.{
            .name = "archerdb-zig",
            .root_source_file = .{ .cwd_relative = "client.zig" },
            .target = target,
            .optimize = optimize,
        });
        b.installArtifact(lib);

        // Unit tests
        const unit_tests = b.addTest(.{
            .root_source_file = .{ .cwd_relative = "tests/unit/client_test.zig" },
            .target = target,
            .optimize = optimize,
        });
        const run_unit_tests = b.addRunArtifact(unit_tests);
        const unit_test_step = b.step("test:unit", "Run unit tests");
        unit_test_step.dependOn(&run_unit_tests.step);

        // Integration tests
        const integration_tests = b.addTest(.{
            .root_source_file = .{ .cwd_relative = "tests/integration/roundtrip_test.zig" },
            .target = target,
            .optimize = optimize,
        });
        const run_integration_tests = b.addRunArtifact(integration_tests);
        const integration_test_step = b.step("test:integration", "Run integration tests (requires running server)");
        integration_test_step.dependOn(&run_integration_tests.step);

        // All tests
        const test_step = b.step("test", "Run all tests");
        test_step.dependOn(&run_unit_tests.step);
        test_step.dependOn(&run_integration_tests.step);
    }
    ```

    ```zig
    // tests/integration/roundtrip_test.zig
    // Integration tests for Zig SDK against real ArcherDB server
    // Requires: server running at http://localhost:3001 (use Phase 11 cluster harness)

    const std = @import("std");
    const Client = @import("../../client.zig").Client;
    const types = @import("../../types.zig");

    // Test helper to get server URL (default or from env)
    fn getServerUrl() []const u8 {
        return std.process.getEnvVarOwned(std.heap.page_allocator, "ARCHERDB_URL") catch "http://localhost:3001";
    }

    test "integration: ping returns pong" {
        var client = try Client.init(std.testing.allocator, getServerUrl());
        defer client.deinit();
        const result = try client.ping();
        try std.testing.expect(result == true);
    }

    test "integration: insert and query roundtrip" {
        var client = try Client.init(std.testing.allocator, getServerUrl());
        defer client.deinit();

        // Insert event
        const events = [_]types.GeoEvent{.{
            .entity_id = 0x12345678_12345678_12345678_12345678,
            .lat_nano = types.degreesToNano(37.7749),
            .lon_nano = types.degreesToNano(-122.4194),
            .group_id = 1,
        }};
        const insert_result = try client.insertEvents(std.testing.allocator, &events);
        defer std.testing.allocator.free(insert_result);

        // Query back by UUID
        const queried = try client.getLatestByUUID(std.testing.allocator, events[0].entity_id);
        try std.testing.expect(queried != null);
        try std.testing.expectEqual(events[0].lat_nano, queried.?.lat_nano);

        // Clean up
        _ = try client.deleteEntities(std.testing.allocator, &[_]u128{events[0].entity_id});
    }
    ```
