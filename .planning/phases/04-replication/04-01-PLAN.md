- Headers must be lowercase, sorted alphabetically
- Header values trimmed of leading/trailing whitespace
- URI must be URI-encoded (except `/`)
- Query parameters sorted by key name

```zig
fn shipImpl(ctx, entry, body) TransportError!void {
    // Build S3 object key: {prefix}/ship/{op:020}.wal
    const key = try std.fmt.allocPrint(allocator, "{s}/ship/{d:0>20}.wal", .{self.prefix, entry.op});
    defer allocator.free(key);

    // Build content: header (64 bytes) + body
    const content = try buildContent(entry, body);
    defer allocator.free(content);

    // Calculate Content-MD5
    var md5_hash: [16]u8 = undefined;
    std.crypto.hash.Md5.hash(content, &md5_hash, .{});
    const content_md5 = std.base64.standard.encode(&md5_hash);

    // Upload with retry
    try self.uploadWithRetry(key, content, &content_md5);
}
```

4. Add retry logic per CONTEXT.md decisions:
```zig
fn uploadWithRetry(self, key, content, content_md5) !void {
    const max_retries: u32 = 10;
    var retry: u32 = 0;

    while (retry < max_retries) : (retry += 1) {
        if (content.len >= self.multipart_threshold) {
            self.s3_client.?.multipartUpload(self.bucket, key, content) catch |err| {
                log.warn("Multipart upload failed (retry {d}): {}", .{retry, err});
                const delay = calculateBackoff(retry, &self.retry_prng);
                std.time.sleep(delay * std.time.ns_per_ms);
                continue;
            };
        } else {
            self.s3_client.?.putObject(self.bucket, key, content, content_md5) catch |err| {
                log.warn("S3 upload failed (retry {d}): {}", .{retry, err});
                const delay = calculateBackoff(retry, &self.retry_prng);
                std.time.sleep(delay * std.time.ns_per_ms);
                continue;
            };
        }
        return; // Success
    }
    return error.UploadFailed; // All retries exhausted -> caller should spill to disk
}

fn calculateBackoff(retry_count: u32, prng: *std.rand.DefaultPrng) u64 {
    // Exponential: 1s, 2s, 4s, 8s... capped at 512s (~8.5 min)
    const base_ms: u64 = 1000;
    const max_ms: u64 = 512_000;
    const delay = @min(base_ms << @intCast(retry_count), max_ms);

    // Add jitter: +/- 25%
    const jitter_range: i64 = @intCast(delay / 4);
    const jitter = prng.random().intRangeAtMost(i64, -jitter_range, jitter_range);
    return @intCast(@max(100, @as(i64, @intCast(delay)) + jitter));
}
```
