const std = @import("std");
const s3_client = @import("s3_client.zig");
const spillover = @import("spillover.zig");
const replication = @import("../replication.zig");

/// MinIO test context - manages Docker container lifecycle
const MinioTestContext = struct {
    allocator: std.mem.Allocator,
    container_id: []const u8,
    endpoint: []const u8,
    bucket: []const u8,
    credentials: s3_client.S3Client.Credentials,

    const default_endpoint = "http://127.0.0.1:9000";
    const default_access_key = "minioadmin";
    const default_secret_key = "minioadmin";
    const test_bucket = "test-replication";

    /// Start MinIO Docker container
    pub fn start(allocator: std.mem.Allocator) !MinioTestContext {
        // Check if MinIO is already running (for local dev)
        if (isMinioRunning(allocator)) {
            return .{
                .allocator = allocator,
                .container_id = "",  // External container
                .endpoint = default_endpoint,
                .bucket = test_bucket,
                .credentials = .{
                    .access_key_id = default_access_key,
                    .secret_access_key = default_secret_key,
                },
            };
        }

        // Start container
        const result = try std.process.Child.run(.{
            .allocator = allocator,
            .argv = &[_][]const u8{
                "docker", "run", "-d", "--rm",
                "-p", "9000:9000",
                "-p", "9001:9001",
                "-e", "MINIO_ROOT_USER=" ++ default_access_key,
                "-e", "MINIO_ROOT_PASSWORD=" ++ default_secret_key,
                "minio/minio", "server", "/data", "--console-address", ":9001",
            },
        });
        defer allocator.free(result.stdout);
        defer allocator.free(result.stderr);

        // Get container ID (first line of stdout)
        const container_id = std.mem.trim(u8, result.stdout, "\n ");
        const id_copy = try allocator.dupe(u8, container_id);

        // Wait for MinIO to be ready
        var ready = false;
        for (0..30) |_| {  // 30 second timeout
            std.time.sleep(1 * std.time.ns_per_s);
            if (isMinioRunning(allocator)) {
                ready = true;
                break;
            }
        }

        if (!ready) {
            // Cleanup on failure
            _ = std.process.Child.run(.{
                .allocator = allocator,
                .argv = &[_][]const u8{"docker", "stop", id_copy},
            }) catch {};
            allocator.free(id_copy);
            return error.MinioStartupTimeout;
        }

        const ctx = MinioTestContext{
            .allocator = allocator,
            .container_id = id_copy,
            .endpoint = default_endpoint,
            .bucket = test_bucket,
            .credentials = .{
                .access_key_id = default_access_key,
                .secret_access_key = default_secret_key,
            },
        };

        // Create test bucket
        try ctx.createBucket();

        return ctx;
    }

    /// Stop MinIO Docker container
    pub fn stop(self: *MinioTestContext) void {
        if (self.container_id.len > 0) {
            _ = std.process.Child.run(.{
                .allocator = self.allocator,
                .argv = &[_][]const u8{"docker", "stop", self.container_id},
            }) catch {};
            self.allocator.free(self.container_id);
        }
    }

    /// Create the test bucket via mc client
    fn createBucket(self: MinioTestContext) !void {
        // Use S3 API to create bucket
        var client = try s3_client.S3Client.init(self.allocator, .{
            .endpoint = self.endpoint,
            .region = "us-east-1",
            .credentials = self.credentials,
        });
        defer client.deinit();

        client.createBucket(self.bucket) catch |err| switch (err) {
            error.BucketAlreadyExists => {},  // OK
            else => return err,
        };
    }

    /// Check if MinIO is running
    fn isMinioRunning(allocator: std.mem.Allocator) bool {
        // Try to connect to MinIO health endpoint
        var client = std.http.Client{ .allocator = allocator };
        defer client.deinit();

        const uri = std.Uri.parse("http://127.0.0.1:9000/minio/health/live") catch return false;
        var req = client.request(.GET, uri, .{}, .{}) catch return false;
        defer req.deinit();
        req.wait() catch return false;

        return req.response.status == .ok;
    }

    /// Get an S3 client configured for this MinIO instance
    pub fn getClient(self: MinioTestContext) !s3_client.S3Client {
        return s3_client.S3Client.init(self.allocator, .{
            .endpoint = self.endpoint,
            .region = "us-east-1",
            .credentials = self.credentials,
        });
    }
};

/// Skip test if Docker not available
fn skipIfNoDocker() bool {
    const result = std.process.Child.run(.{
        .allocator = std.testing.allocator,
        .argv = &[_][]const u8{"docker", "version"},
    }) catch return true;
    std.testing.allocator.free(result.stdout);
    std.testing.allocator.free(result.stderr);
    return result.term.Exited != 0;
}

test "S3 upload: single object upload to MinIO" {
    if (skipIfNoDocker()) {
        std.debug.print("Skipping: Docker not available\n", .{});
        return;
    }

    const allocator = std.testing.allocator;
    var minio = try MinioTestContext.start(allocator);
    defer minio.stop();

    var client = try minio.getClient();
    defer client.deinit();

    // Upload test object
    const key = "test/upload.txt";
    const body = "Hello, MinIO!";
    const result = try client.putObject(minio.bucket, key, body, null);

    try std.testing.expect(result.etag.len > 0);

    // Verify object exists by reading it back
    const downloaded = try client.getObject(minio.bucket, key);
    defer allocator.free(downloaded);
    try std.testing.expectEqualStrings(body, downloaded);
}

test "S3 upload: Content-MD5 verification" {
    if (skipIfNoDocker()) return;

    const allocator = std.testing.allocator;
    var minio = try MinioTestContext.start(allocator);
    defer minio.stop();

    var client = try minio.getClient();
    defer client.deinit();

    const body = "Content to verify";

    // Calculate MD5
    var md5_hash: [16]u8 = undefined;
    std.crypto.hash.Md5.hash(body, &md5_hash, .{});
    var md5_base64: [24]u8 = undefined;
    _ = std.base64.standard.Encoder.encode(&md5_base64, &md5_hash);

test "S3 upload: multipart upload for large file" {
    if (skipIfNoDocker()) return;

    const allocator = std.testing.allocator;
    var minio = try MinioTestContext.start(allocator);
    defer minio.stop();

    var client = try minio.getClient();
    defer client.deinit();

    // Create 20MB test data (two 10MB parts, under 100MB threshold but tests multipart)
    const part_size: usize = 10 * 1024 * 1024;  // 10MB
    const data = try allocator.alloc(u8, part_size * 2);
    defer allocator.free(data);
    @memset(data, 'X');

    // Initiate multipart
    const upload_id = try client.initiateMultipartUpload(minio.bucket, "large-file.bin");
    defer allocator.free(upload_id);

    // Upload parts
    var parts: [2]s3_client.PartInfo = undefined;
    parts[0] = .{
        .part_number = 1,
        .etag = try client.uploadPart(minio.bucket, "large-file.bin", upload_id, 1, data[0..part_size]),
    };
    defer allocator.free(parts[0].etag);

    parts[1] = .{
        .part_number = 2,
        .etag = try client.uploadPart(minio.bucket, "large-file.bin", upload_id, 2, data[part_size..]),
    };
    defer allocator.free(parts[1].etag);

    // Complete multipart
    try client.completeMultipartUpload(minio.bucket, "large-file.bin", upload_id, &parts);

    // Verify file size
    const stat = try client.headObject(minio.bucket, "large-file.bin");
    try std.testing.expectEqual(@as(u64, part_size * 2), stat.content_length);
}

test "S3 upload: S3RelayTransport ships to MinIO" {
    if (skipIfNoDocker()) return;

    try t.ship(&entry, "HELLO");

    defer client.deinit();

    const key = "replication/ship/00000000000000000042.wal";

test "spillover: write and recover entries" {
    const allocator = std.testing.allocator;

    // Create temp directory
    var tmp = std.testing.tmpDir(.{});
    defer tmp.cleanup();
    const tmp_path = try tmp.dir.realpathAlloc(allocator, ".");
    defer allocator.free(tmp_path);

    // Create spillover manager
    var sm = try spillover.SpilloverManager.init(allocator, tmp_path);
    defer sm.deinit();

    // Create test entries
    var entries: [3]spillover.SpillEntry = undefined;
    for (&entries, 0..) |*entry, i| {
        var header = replication.ShipEntry{
            .op = @intCast(i + 1),
            .commit_timestamp_ns = @intCast(std.time.nanoTimestamp()),
            .body_size = 5,
            .primary_region_id = 1,
        };
        header.setChecksum("test1");
        entry.* = .{
            .header = header,
            .body = "test1",
        };
    }

    // Spill entries
    try sm.spillEntries(&entries);

    // Verify metadata
    try std.testing.expectEqual(@as(u32, 1), sm.meta.segment_count);
    try std.testing.expect(sm.meta.total_bytes > 0);

    // Deinit and reinit to simulate restart
    sm.deinit();
    sm = try spillover.SpilloverManager.init(allocator, tmp_path);

    // Recover entries
    var iter = try sm.recoverEntries();
    defer iter.deinit();

    var count: u32 = 0;
    while (iter.next()) |_| {
        count += 1;
    }
    try std.testing.expectEqual(@as(u32, 3), count);
}

test "spillover: atomic write survives crash simulation" {
    const allocator = std.testing.allocator;

    var tmp = std.testing.tmpDir(.{});
    defer tmp.cleanup();
    const tmp_path = try tmp.dir.realpathAlloc(allocator, ".");
    defer allocator.free(tmp_path);

    // Create spillover directory
    const spillover_dir = try std.fs.path.join(allocator, &.{tmp_path, "spillover"});
    defer allocator.free(spillover_dir);
    try std.fs.cwd().makePath(spillover_dir);

    // Simulate partial write (create temp file but don't rename)
    const temp_file = try std.fs.path.join(allocator, &.{spillover_dir, ".tmp_000001.spill"});
    defer allocator.free(temp_file);
    var file = try std.fs.cwd().createFile(temp_file, .{});
    try file.writeAll("partial garbage data");
    file.close();

    // Init should handle partial file gracefully
    var sm = try spillover.SpilloverManager.init(allocator, tmp_path);
    defer sm.deinit();

    // Should have no valid entries
    try std.testing.expect(!sm.hasPending());
}

test "spillover: ShipQueue spillover and recovery integration" {
    const allocator = std.testing.allocator;

    var tmp = std.testing.tmpDir(.{});
    defer tmp.cleanup();
    const tmp_path = try tmp.dir.realpathAlloc(allocator, ".");
    defer allocator.free(tmp_path);

    // Create queue with small memory limit to force spillover
    var queue = try replication.ShipQueue.init(allocator, .{
        .memory_max = 2,  // Very small to force spillover

        .spillover_dir = tmp_path,

    // Should have recovered any spilled entries

    _ = recovered;

test "spillover: end-to-end with S3 failure and recovery" {
    if (skipIfNoDocker()) return;

    const allocator = std.testing.allocator;

    var tmp = std.testing.tmpDir(.{});
    defer tmp.cleanup();
    const tmp_path = try tmp.dir.realpathAlloc(allocator, ".");
    defer allocator.free(tmp_path);

    var minio = try MinioTestContext.start(allocator);
    defer minio.stop();

    // Create coordinator with small retry count to force spillover quickly
    var coordinator = try replication.ShipCoordinator.init(allocator, .{
        .primary_region_id = 1,
        .max_retries = 1,  // Fail fast for testing
        .data_dir = tmp_path,
        .queue_config = .{
            .memory_max = 100,
            .spillover_dir = tmp_path,
        },
    });
    defer coordinator.deinit();

    // Add S3 follower
    try coordinator.addFollower(.{
        .region_id = 2,
        .name = "test-follower",
        .transport_type = .s3_relay,
        .s3_bucket = minio.bucket,
        .s3_prefix = "replication",
    });

    // Queue some entries
    try coordinator.queueCommit(1, @intCast(std.time.nanoTimestamp()), "body1");
    try coordinator.queueCommit(2, @intCast(std.time.nanoTimestamp()), "body2");

    // Pause MinIO to cause failures
    _ = try std.process.Child.run(.{
        .allocator = allocator,
        .argv = &[_][]const u8{"docker", "pause", minio.container_id},
    });

    // Try to ship (will fail and trigger spillover)
    coordinator.tick() catch {};
    coordinator.tick() catch {};  // Second attempt should spill

    // Check spillover happened
    try std.testing.expect(coordinator.queue.stats.disk_bytes > 0);

    // Unpause MinIO
    _ = try std.process.Child.run(.{
        .allocator = allocator,
        .argv = &[_][]const u8{"docker", "unpause", minio.container_id},
    });
    std.time.sleep(2 * std.time.ns_per_s);

    // Recover and retry
    _ = try coordinator.queue.recoverFromDisk();

    // Now shipping should succeed
    try coordinator.tick();

    // Verify uploaded
    var client = try minio.getClient();
    defer client.deinit();

    // At least one entry should be uploaded
    const key1 = "replication/ship/00000000000000000001.wal";
    _ = client.headObject(minio.bucket, key1) catch |err| {
        // If first entry not there, second might be
        const key2 = "replication/ship/00000000000000000002.wal";
        _ = try client.headObject(minio.bucket, key2);
    };
}
