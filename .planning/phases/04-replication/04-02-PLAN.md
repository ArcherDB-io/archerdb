```zig
/// Spillover segment header (binary, fixed 64 bytes)
pub const SpilloverSegment = extern struct {
    magic: [4]u8 = .{ 'S', 'P', 'I', 'L' },
    version: u16 = 1,
    entry_count: u32,
    total_bytes: u64,
    first_op: u64,
    last_op: u64,
    checksum: u128,  // CRC of all entries in segment
    _padding: [14]u8 = .{0} ** 14,

    pub const size = 64;
};

/// Manages disk spillover
pub const SpilloverManager = struct {
    allocator: Allocator,
    spillover_dir: []const u8,
    meta: SpilloverMeta,
    current_segment: ?std.fs.File,
    current_segment_id: u32,
    entries_in_current: u32,
    max_entries_per_segment: u32 = 1000,

    pub fn init(allocator: Allocator, data_dir: []const u8) !SpilloverManager;
    pub fn deinit(self: *SpilloverManager) void;

    /// Spill entries to disk (atomic write: temp file + rename)
    pub fn spillEntries(self: *SpilloverManager, entries: []const SpillEntry) !void;

    /// Recover entries from disk (returns iterator)
    pub fn recoverEntries(self: *SpilloverManager) !EntryIterator;

    /// Mark entries as uploaded (delete spillover files)
    pub fn markUploaded(self: *SpilloverManager, through_op: u64) !void;

    /// Get total bytes on disk
    pub fn getDiskBytes(self: *SpilloverManager) u64;

    /// Check if there are pending spillover entries
    pub fn hasPending(self: *SpilloverManager) bool;
};

pub const SpillEntry = struct {
```

```zig
fn writeSegment(self, entries) !void {
    // 1. Write to temp file
    const temp_path = try std.fmt.allocPrint(self.allocator, "{s}/.tmp_{d}.spill", .{self.spillover_dir, self.current_segment_id});
    defer self.allocator.free(temp_path);

    var file = try std.fs.cwd().createFile(temp_path, .{});
    defer file.close();

    // Write segment header
    var header = SpilloverSegment{...};
    try file.writeAll(std.mem.asBytes(&header));

    // Write entries
    for (entries) |entry| {
        try file.writeAll(std.mem.asBytes(&entry.header));
        try file.writeAll(entry.body);
    }

    // Sync to disk
    try file.sync();

    // 2. Atomic rename
    const final_path = try std.fmt.allocPrint(self.allocator, "{s}/{d:0>6}.spill", .{self.spillover_dir, self.current_segment_id});
    defer self.allocator.free(final_path);
    try std.fs.cwd().rename(temp_path, final_path);

    // 3. Update and persist metadata
    self.meta.segment_count += 1;
    try self.persistMeta();
}
```

```zig
fn spillToDisk(self: *ShipQueue) !void {
    if (self.spillover_manager == null) return error.NoSpilloverPath;

    // Spill half of memory queue to disk (batch spillover)
    const entries_to_spill = self.memory_queue.items.len / 2;
    if (entries_to_spill == 0) return;

    // Build SpillEntry array
    var spill_entries = try self.allocator.alloc(spillover.SpillEntry, entries_to_spill);
    defer self.allocator.free(spill_entries);

    for (0..entries_to_spill) |i| {
        const entry = self.memory_queue.items[i];
        spill_entries[i] = .{
            .header = entry.header,
            .body = entry.body,
        };
    }

    // Spill to disk (atomic)
    try self.spillover_manager.?.spillEntries(spill_entries);

    // Remove from memory queue
    for (0..entries_to_spill) |_| {
        const removed = self.memory_queue.orderedRemove(0);
        self.allocator.free(removed.body);
        self.stats.memory_entries -|= 1;
        self.stats.memory_bytes -|= removed.body.len;
    }

    // Update metrics
    self.stats.disk_entries += entries_to_spill;
    self.stats.disk_bytes = self.spillover_manager.?.getDiskBytes();

    // Update Prometheus metrics
    const Registry = @import("archerdb/metrics.zig").Registry;
    Registry.replication_spillover_bytes.store(self.stats.disk_bytes, .monotonic);
}
```

```zig
pub fn recoverFromDisk(self: *ShipQueue) !u64 {
    if (self.spillover_manager == null) return 0;

    var recovered: u64 = 0;
    var iter = try self.spillover_manager.?.recoverEntries();
    defer iter.deinit();

    while (iter.next()) |entry| {
        if (self.memory_queue.items.len >= self.memory_max) {
            log.warn("Spillover recovery paused (memory queue full)", .{});
            break;
        }

        // Copy body
        const body_copy = try self.allocator.dupe(u8, entry.body);
        try self.memory_queue.append(.{
            .header = entry.header,
            .body = body_copy,
            .queued_at_ns = @intCast(std.time.nanoTimestamp()),
            .retry_count = 0,
        });

        recovered += 1;
        self.stats.memory_entries += 1;
        self.stats.memory_bytes += entry.body.len;
    }

    if (recovered > 0) {
        log.info("Recovered {} entries from disk spillover", .{recovered});
    }

    return recovered;
}
```

```zig
pub fn tick(self: *ShipCoordinator) !void {
    const now = @as(u64, @intCast(std.time.nanoTimestamp()));
    const Registry = @import("archerdb/metrics.zig").Registry;

    // Process queue for each follower
    for (self.followers.items) |*follower| {
        // ... existing backoff check ...

        // Ship entries
        while (self.queue.peek()) |entry| {
            const transport = follower.transport.?;

            transport.ship(&entry.header, entry.body) catch |err| {
                log.warn("Ship failed to region {}: {}", .{follower.config.region_id, err});
                follower.consecutive_failures += 1;

                // Check if we should spill to disk
                if (follower.consecutive_failures >= self.max_retries) {
                    log.warn("Max retries exceeded, spilling to disk", .{});

                    // Spill current queue to disk
                    if (self.queue.spillover_manager != null) {
                        self.queue.spillToDisk() catch |spill_err| {
                            log.err("Spillover failed: {}", .{spill_err});
                        };

                        // Update replication state to degraded
                        Registry.replication_state.store(1, .monotonic);
                    }
                }

                // Exponential backoff
                follower.retry_backoff_ns = @min(
                    follower.retry_backoff_ns * 2,
                    repl_constants.ship_retry_backoff_max_ms * std.time.ns_per_ms,
                );
                follower.last_retry_ns = now;
                break;
            };

            // Success
            const dequeued = self.queue.dequeue().?;

            // Mark as uploaded in spillover manager (cleans up disk files)
            if (self.queue.spillover_manager) |*sm| {
                sm.markUploaded(entry.header.op) catch {};
            }

            self.queue.markShipped(dequeued);

            follower.lag.shipped_op = entry.header.op;
            follower.lag.last_ship_ns = now;
            follower.consecutive_failures = 0;
            follower.retry_backoff_ns =
                repl_constants.ship_retry_backoff_initial_ms * std.time.ns_per_ms;

            // If spillover cleared, set state back to healthy
            if (self.queue.spillover_manager) |sm| {
                if (!sm.hasPending()) {
                    Registry.replication_state.store(0, .monotonic);
                }
            }

            // Update queue depth metric
            const follower_idx = @as(usize, @intCast(follower.config.region_id % Registry.max_followers));
            Registry.replication_ship_queue_depth[follower_idx].store(self.queue.depth(), .monotonic);
        }
    }

    // Update lag metrics
    self.updateLagMetrics();
}
```

```zig
fn updateLagMetrics(self: *ShipCoordinator) void {
    const Registry = @import("archerdb/metrics.zig").Registry;

    // Calculate oldest unuploaded timestamp
    if (self.queue.peek()) |oldest| {
        const now_ns: u64 = @intCast(std.time.nanoTimestamp());
        const lag_ns = now_ns -| oldest.queued_at_ns;

        // This is time-based lag (seconds since oldest unuploaded)
        Registry.replication_lag_ns.store(lag_ns, .monotonic);
        Registry.replication_lag_ops.store(self.queue.depth(), .monotonic);
    } else {
        Registry.replication_lag_ns.store(0, .monotonic);
        Registry.replication_lag_ops.store(0, .monotonic);
    }
}
```
