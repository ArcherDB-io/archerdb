**MULTI-02: Leader election within 5 seconds test**
The existing test patterns show leader election is implicitly tested through view changes. Create an explicit timing verification test:
```zig
test "MULTI-02: leader election completes within 5 seconds" {
    const t = try TestContext.init(.{ .replica_count = 3 });
    defer t.deinit();

    var c = t.clients(.{});
    try c.request(2, 2);  // Establish baseline

    // Kill the primary
    t.replica(.A0).stop();

    // Run until new primary elected or timeout
    // tick_ms = 10, so 5000ms = 500 ticks
    const tick_limit = 5000 / constants.tick_ms;
    var ticks: u64 = 0;

    while (ticks < tick_limit) : (ticks += 1) {
        _ = t.tick();
        // Check if cluster recovered (any remaining replica is primary and normal)
        if (can_commit(t)) break;
    }

    // Verify election completed within time limit
    try expect(ticks < tick_limit);

    // Verify new primary can accept requests
    try c.request(3, 3);
}
```

**MULTI-03: Replica rejoin and catchup test**
```zig
test "MULTI-03: crashed replica rejoins and catches up" {
    const t = try TestContext.init(.{ .replica_count = 3 });
    defer t.deinit();

    var c = t.clients(.{});
    try c.request(5, 5);

    var b2 = t.replica(.B2);
    b2.stop();  // Crash one replica

    // Cluster continues without it
    try c.request(10, 10);
    try expectEqual(t.replica(.A0).commit(), 10);
    try expectEqual(t.replica(.B1).commit(), 10);

    // Replica rejoins
    try b2.open();
    t.run();  // Let it catch up

    // Verify it caught up
    try expectEqual(b2.status(), .normal);
    try expectEqual(b2.commit(), 10);
}
```
