   ```zig
   pub fn insert(self: *Self, entity_id: u128, entry: Entry) !void {
       var current = entry;
       var current_id = entity_id;
       var displacement_count: u32 = 0;

       while (displacement_count < max_displacement) {
           // Try primary slot
           const s1 = self.slot1(current_id);
           if (self.entries[s1].is_empty() or self.entries[s1].is_tombstone()) {
               self.entries[s1] = current;
               self.count.fetchAdd(1, .monotonic);
               return;
           }

           // Try secondary slot
           const s2 = self.slot2(current_id);
           if (self.entries[s2].is_empty() or self.entries[s2].is_tombstone()) {
               self.entries[s2] = current;
               self.count.fetchAdd(1, .monotonic);
               return;
           }

           // Both full - displace from primary slot
           const displaced = self.entries[s1];
           self.entries[s1] = current;
           current = displaced;
           current_id = displaced.entity_id;
           displacement_count += 1;
       }

       // Max displacement reached - table too full
       return error.TableFull;
   }
   ```

2. For in-place updates (same entity_id, newer timestamp), check both slots first:
   - If found at s1 or s2, update in place (LWW semantics)
   - Only do displacement if entity not already present

3. Add error.TableFull to error set if not present.
