```yaml
# RAM Index Memory Alert Rules
# Following patterns from storage.yml

groups:
  - name: ram-index-load
    rules:
      - alert: RamIndexLoadFactorHigh
        expr: archerdb_ram_index_load_factor > 700
        for: 10m
        labels:
          severity: warning
        annotations:
          summary: "RAM index load factor exceeds 70%"
          description: "Load factor is {{ $value / 10 }}%. Cuckoo hashing performance degrades above 70%. Consider increasing capacity."
          runbook_url: "https://archerdb.io/docs/ops/runbook#ram-index-load-factor"

      - alert: RamIndexLoadFactorCritical
        expr: archerdb_ram_index_load_factor > 800
        for: 5m
        labels:
          severity: critical
        annotations:
          summary: "RAM index load factor exceeds 80%"
          description: "Load factor is {{ $value / 10 }}%. Index is approaching failure threshold. Immediate action required."
          runbook_url: "https://archerdb.io/docs/ops/runbook#ram-index-load-factor"

  - name: ram-index-memory
    rules:
      - alert: RamIndexMemoryHigh
        expr: archerdb_ram_index_memory_bytes > 100 * 1024 * 1024 * 1024
        for: 5m
        labels:
          severity: info
        annotations:
          summary: "RAM index using over 100 GiB"
          description: "RAM index is using {{ $value | humanize1024 }}. Monitor system memory utilization."
          runbook_url: "https://archerdb.io/docs/ops/runbook#ram-index-memory"

  - name: ram-index-lookup
    rules:
      - alert: RamIndexHitRateLow
        expr: |
          (
            rate(archerdb_ram_index_lookup_hits_total[5m]) /
            rate(archerdb_ram_index_lookups_total[5m])
          ) < 0.5
        for: 15m
        labels:
          severity: warning
        annotations:
          summary: "RAM index lookup hit rate below 50%"
          description: "Hit rate is {{ $value | humanizePercentage }}. High miss rate may indicate entities not being found or stale lookups."
          runbook_url: "https://archerdb.io/docs/ops/runbook#ram-index-hit-rate"

  - name: ram-index-insert
    rules:
      - alert: RamIndexHighDisplacements
        expr: rate(archerdb_ram_index_displacements_total[5m]) > 1000
        for: 10m
        labels:
          severity: warning
        annotations:
          summary: "RAM index cuckoo displacement rate high"
          description: "Displacement rate is {{ $value }}/s. High displacement indicates load factor too high or poor hash distribution."
          runbook_url: "https://archerdb.io/docs/ops/runbook#ram-index-displacements"

      - alert: RamIndexInsertFailures
        expr: increase(archerdb_ram_index_insert_failures_total[5m]) > 0
        for: 1m
        labels:
          severity: critical
        annotations:
          summary: "RAM index insert failures detected"
          description: "Insert operations are failing. Index may be full or encountering displacement loops."
          runbook_url: "https://archerdb.io/docs/ops/runbook#ram-index-insert-failure"
```
