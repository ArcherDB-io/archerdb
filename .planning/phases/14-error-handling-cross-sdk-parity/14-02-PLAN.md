   ```python
   """Cross-SDK parity test runner.

   Runs identical operations across all 5 SDKs and verifies results match.
   Per CONTEXT.md: Python SDK as golden reference, server as ultimate truth.
   """
   import os
   import sys
   import json
   import subprocess
   from pathlib import Path
   from typing import Dict, List, Any, Optional
   from dataclasses import dataclass, asdict
   from datetime import datetime

   # Add test infrastructure to path
   sys.path.insert(0, str(Path(__file__).parent.parent.parent))
   from test_infrastructure.harness import ArcherDBCluster, ClusterConfig
   from test_infrastructure.fixtures import load_fixture

   from parity_verifier import ParityVerifier, ParityResult
   from sdk_runners import python_runner, node_runner, go_runner, java_runner, c_runner, zig_runner

   SDK_RUNNERS = {
       "python": python_runner,
       "node": node_runner,
       "go": go_runner,
       "java": java_runner,
       "c": c_runner,
       "zig": zig_runner,
   }

   OPERATIONS = [
       "insert", "upsert", "delete",
       "query-uuid", "query-uuid-batch", "query-radius", "query-polygon", "query-latest",
       "ping", "status", "topology",
       "ttl-set", "ttl-extend", "ttl-clear"
   ]

   def run_parity_tests(
       server_url: str,
       operations: Optional[List[str]] = None,
       sdks: Optional[List[str]] = None
   ) -> List[ParityResult]:
       """Run parity tests across all SDKs for all operations."""
       operations = operations or OPERATIONS
       sdks = sdks or list(SDK_RUNNERS.keys())

       verifier = ParityVerifier(server_url)
       results = []

       for op in operations:
           # Load fixture for operation
           fixture = load_fixture(op.replace("-", "_"))
           if not fixture:
               print(f"Warning: No fixture for {op}")
               continue

           for test_case in fixture.get("test_cases", []):
               input_data = test_case.get("input", {})
               result = verifier.verify_parity(op, input_data, sdks)
               results.append(result)

       return results

   def main():
       """CLI entry point."""
       import argparse
       parser = argparse.ArgumentParser(description="Run SDK parity tests")
       parser.add_argument("--url", default="http://127.0.0.1:7000")
       parser.add_argument("--ops", nargs="*", help="Operations to test")
       parser.add_argument("--sdks", nargs="*", help="SDKs to test")
       parser.add_argument("--output", default="reports/parity.json")
       args = parser.parse_args()

       results = run_parity_tests(args.url, args.ops, args.sdks)

       # Write JSON report
       verifier = ParityVerifier(args.url)
       verifier.write_reports(results, args.output, "docs/PARITY.md")

   if __name__ == "__main__":
       main()
   ```

3. **tests/parity_tests/parity_verifier.py**: Result comparison
   ```python
   """Cross-SDK result verification with exact nanodegree matching.

   Per CONTEXT.md:
   - Exact match required for coordinates (nanodegrees)
   - No epsilon tolerance
   - Python as golden reference
   """
   import json
   from pathlib import Path
   from typing import Dict, List, Any, Optional
   from dataclasses import dataclass, field
   from datetime import datetime

   @dataclass
   class ParityResult:
       operation: str
       test_case: str
       passed: bool
       sdk_results: Dict[str, Any] = field(default_factory=dict)
       mismatches: List[Dict[str, Any]] = field(default_factory=list)
       error: Optional[str] = None

   class ParityVerifier:
       """Verify SDK results match across all implementations."""

       SDK_ORDER = ["python", "node", "go", "java", "c", "zig"]

       def __init__(self, server_url: str):
           self.server_url = server_url

       def verify_parity(
           self,
           operation: str,
           input_data: dict,
           sdks: List[str]
       ) -> ParityResult:
           """Run operation across SDKs and verify results match."""
           results = {}
           for sdk in sdks:
               try:
                   results[sdk] = self._run_sdk(sdk, operation, input_data)
               except Exception as e:
                   results[sdk] = {"error": str(e)}

           # Python as golden reference (per CONTEXT.md)
           if "python" not in results:
               return ParityResult(
                   operation=operation,
                   test_case=str(input_data),
                   passed=False,
                   error="Python reference not available"
               )

           reference = results["python"]
           mismatches = []

           for sdk in sdks:
               if sdk == "python":
                   continue
               if not self._compare_results(reference, results[sdk]):
                   mismatches.append({
                       "sdk": sdk,
                       "expected": reference,
                       "actual": results[sdk],
                       "diff": self._compute_diff(reference, results[sdk])
                   })

           return ParityResult(
               operation=operation,
               test_case=str(input_data),
               passed=len(mismatches) == 0,
               sdk_results=results,
               mismatches=mismatches
           )

       def _compare_results(self, expected: Any, actual: Any) -> bool:
           """Compare results with exact nanodegree matching (per CONTEXT.md)."""
           if type(expected) != type(actual):
               return False

           if isinstance(expected, dict):
               if set(expected.keys()) != set(actual.keys()):
                   return False
               return all(
                   self._compare_results(expected[k], actual[k])
                   for k in expected.keys()
               )

           if isinstance(expected, list):
               if len(expected) != len(actual):
                   return False
               return all(
                   self._compare_results(e, a)
                   for e, a in zip(expected, actual)
               )

           # Exact match for numbers (no epsilon per CONTEXT.md)
           return expected == actual

       def _compute_diff(self, expected: Any, actual: Any) -> Dict:
           """Compute semantic diff between results."""
           # Use deepdiff if available, else simple comparison
           try:
               from deepdiff import DeepDiff
               return DeepDiff(expected, actual, significant_digits=15).to_dict()
           except ImportError:
               return {"expected": expected, "actual": actual}

       def _run_sdk(self, sdk: str, operation: str, input_data: dict) -> dict:
           """Run operation in specified SDK."""
           from sdk_runners import python_runner, node_runner, go_runner
           from sdk_runners import java_runner, c_runner, zig_runner

           runners = {
               "python": python_runner,
               "node": node_runner,
               "go": go_runner,
               "java": java_runner,
               "c": c_runner,
               "zig": zig_runner,
           }

           runner = runners.get(sdk)
           if not runner:
               raise ValueError(f"Unknown SDK: {sdk}")

           return runner.run_operation(self.server_url, operation, input_data)

       def write_reports(
           self,
           results: List[ParityResult],
           json_path: str,
           markdown_path: str
       ):
           """Generate JSON and Markdown reports (per CONTEXT.md)."""
           # JSON report for CI automation
           json_report = {
               "generated": datetime.utcnow().isoformat(),
               "summary": {
                   "total_cells": len(results),
                   "passed": sum(1 for r in results if r.passed),
                   "failed": sum(1 for r in results if not r.passed)
               },
               "results": [
                   {
                       "operation": r.operation,
                       "test_case": r.test_case,
                       "passed": r.passed,
                       "mismatches": r.mismatches
                   }
                   for r in results
               ]
           }

           Path(json_path).parent.mkdir(parents=True, exist_ok=True)
           with open(json_path, "w") as f:
               json.dump(json_report, f, indent=2)

           # Markdown report for human review
           self._write_markdown(results, markdown_path)

       def _write_markdown(self, results: List[ParityResult], path: str):
           """Generate human-readable parity matrix."""
           lines = [
               "# SDK Parity Matrix",
               "",
               f"Generated: {datetime.utcnow().isoformat()}",
               "",
               "## Summary",
               "",
               f"- Total tests: {len(results)}",
               f"- Passed: {sum(1 for r in results if r.passed)}",
               f"- Failed: {sum(1 for r in results if not r.passed)}",
               "",
               "## Matrix (14 ops x 5 SDKs = 70 cells)",
               "",
               "| Operation | Python | Node.js | Go | Java | C |",
               "|-----------|--------|---------|----|----|---|"
           ]

           # Group results by operation
           by_op = {}
           for r in results:
               if r.operation not in by_op:
                   by_op[r.operation] = {}
               # Assume first result per op for matrix
               for sdk in self.SDK_ORDER:
                   if sdk in r.sdk_results:
                       by_op[r.operation][sdk] = r.passed

           for op in sorted(by_op.keys()):
               row = f"| {op} |"
               for sdk in self.SDK_ORDER:
                   if sdk in by_op.get(op, {}):
                       symbol = "PASS" if by_op[op][sdk] else "FAIL"
                       row += f" {symbol} |"
                   else:
                       row += " - |"
               lines.append(row)

           Path(path).parent.mkdir(parents=True, exist_ok=True)
           with open(path, "w") as f:
               f.write("\n".join(lines))
   ```

   ```python
   from archerdb import GeoClientSync, GeoClientConfig, GeoEvent

   def run_operation(server_url: str, operation: str, input_data: dict) -> dict:
       """Run operation and return result as dict."""
       config = GeoClientConfig(
           cluster_id=0,
           addresses=[server_url.replace("http://", "")]
       )

       with GeoClientSync(config) as client:
           if operation == "ping":
               result = client.ping()
               return {"success": result.success}

           elif operation == "insert":
               events = [
                   GeoEvent(
                       entity_id=e["entity_id"],
                       latitude=e["latitude"],
                       longitude=e["longitude"]
                   )
                   for e in input_data.get("events", [])
               ]
               result = client.insert_events(events)
               return {"count": result.count}

               result = client.query_radius(
                   latitude=input_data["latitude"],
                   longitude=input_data["longitude"],
                   radius_m=input_data["radius_m"]
               )
               return {
                   "count": result.count,
                   "events": [
                       {
                           "entity_id": e.entity_id,
                           "latitude": e.latitude,
                           "longitude": e.longitude
                       }
                       for e in result.events
                   ]
               }
   ```

   ```python
   """Node.js SDK runner for parity tests.

   Runs Node.js SDK operations via subprocess with JSON input/output.
   """
   import subprocess
   import json
   import os
   from pathlib import Path

   # Path to Node.js parity test script
   NODE_SCRIPT = Path(__file__).parent.parent.parent / "sdk_tests" / "node" / "parity_runner.js"

   def run_operation(server_url: str, operation: str, input_data: dict) -> dict:
       """Run Node.js SDK operation via subprocess.

       The parity_runner.js script accepts:
       - ARCHERDB_URL env var for server address
       - operation as first argument
       - input_data as JSON on stdin

       It outputs the result as JSON on stdout.
       """
       env = os.environ.copy()
       env["ARCHERDB_URL"] = server_url

       # Build Node.js parity runner path
       node_script = str(NODE_SCRIPT)

       # Check if script exists, fall back to inline execution if not
       if not os.path.exists(node_script):
           # Create minimal inline runner
           inline_script = f'''
           const {{ GeoClient }} = require("{Path(__file__).parent.parent.parent.parent / "src" / "clients" / "node" / "src"}");
           const input = JSON.parse(require("fs").readFileSync(0, "utf-8"));
           const client = new GeoClient({{ url: process.env.ARCHERDB_URL }});
           const op = process.argv[2];
           // Route to operation...
           console.log(JSON.stringify({{ error: "inline runner not fully implemented" }}));
           '''
           result = subprocess.run(
               ["node", "-e", inline_script, operation],
               input=json.dumps(input_data),
               capture_output=True,
               text=True,
               env=env,
               timeout=30
           )
       else:
           result = subprocess.run(
               ["node", node_script, operation],
               input=json.dumps(input_data),
               capture_output=True,
               text=True,
               env=env,
               timeout=30
           )

       if result.returncode != 0:
           return {"error": result.stderr or "Node.js runner failed"}

       try:
           return json.loads(result.stdout)
       except json.JSONDecodeError:
           return {"error": f"Invalid JSON output: {result.stdout[:200]}"}
   ```
