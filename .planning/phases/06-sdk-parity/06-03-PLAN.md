package com.archerdb.geo;

import java.util.List;
import java.util.Map;
import java.util.concurrent.CompletableFuture;
import java.util.concurrent.Executor;
import java.util.concurrent.ForkJoinPool;

/**
 * Asynchronous ArcherDB geospatial client using CompletableFuture.
 *
 * <p>Provides non-blocking versions of all GeoClient operations.
 * Operations are executed on the provided Executor (default: ForkJoinPool.commonPool).
 *
 * <p>Example usage:
 * <pre>{@code
 * GeoClientAsync client = GeoClientAsync.create(0L, "127.0.0.1:3001");
 *
 * client.insertEventsAsync(events)
 *     .thenAccept(errors -> {
 *         if (errors.isEmpty()) {
 *             System.out.println("All events inserted!");
 *         }
 *     })
 *     .exceptionally(ex -> {
 *         System.err.println("Insert failed: " + ex);
 *         return null;
 *     });
 * }</pre>
 *
 * @see GeoClient for synchronous operations
 */
public class GeoClientAsync implements AutoCloseable {
```

    private GeoClientAsync(GeoClient delegate, Executor executor) {

    public static GeoClientAsync create(long clusterId, String... addresses) {
        return new GeoClientAsync(

    public static GeoClientAsync create(GeoClient client, Executor executor) {

## Async Support

For async operations using CompletableFuture:

```java
import com.archerdb.geo.GeoClientAsync;

try (GeoClientAsync client = GeoClientAsync.create(0L, "127.0.0.1:3001")) {
    // Insert events asynchronously
    client.insertEventsAsync(events)
        .thenAccept(errors -> {
            if (errors.isEmpty()) {
                System.out.println("All events inserted!");
            }
        })
        .exceptionally(ex -> {
            System.err.println("Insert failed: " + ex.getMessage());
            return null;
        });

    // Query asynchronously
    CompletableFuture<QueryResult> future = client.queryRadiusAsync(filter)
        .thenApply(result -> {
            System.out.println("Found " + result.getEvents().size() + " events");
            return result;
        });

    // Wait for result if needed
    QueryResult result = future.join();

    // Combine multiple async operations
    CompletableFuture.allOf(
        client.queryRadiusAsync(filter1),
        client.queryRadiusAsync(filter2),
        client.queryRadiusAsync(filter3)
    ).join();
}
```

## Exception Handling

ArcherDB uses a typed exception hierarchy:

```java
try {
    QueryResult result = client.queryRadius(filter);
} catch (ValidationException e) {
    // Invalid input (e.g., coordinates out of range)
    System.err.println("Invalid query: " + e.getMessage());
} catch (ConnectionException e) {
    // Network issues - may be retryable
    if (e.isRetryable()) {
        // Implement retry logic
    }
} catch (OperationException e) {
    // Operation failed (timeout, etc.)
    System.err.println("Operation failed: " + e.getErrorCode());
} catch (ArcherDBException e) {
    // Catch-all for other ArcherDB errors
    System.err.println("ArcherDB error: " + e.getMessage());
}
```

| Exception | Error Codes | Retryable | Description |

| OperationException | 4001-4003 | Varies | Operation-level failures |
