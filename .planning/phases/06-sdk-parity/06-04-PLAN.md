/**
 * Base class for all ArcherDB errors.
 *
 * All errors include a numeric code for programmatic handling
 * and a retryable flag indicating if the operation can be retried.
 *
 * @example
 * ```typescript
 * try {
 *   await client.queryRadius(filter)
 * } catch (error) {
 *   if (error instanceof ArcherDBError) {
 *     console.log(`Error code: ${error.code}`)
 *     if (error.retryable) {
 *       // Implement retry logic
 *     }
 *   }
 * }
 * ```
 */
export class ArcherDBError extends Error {
  /** Numeric error code for programmatic handling */
  readonly code: number
  /** Whether this error may succeed on retry */
  readonly retryable: boolean

  constructor(code: number, message: string, retryable: boolean) {
    super(message)
    this.name = 'ArcherDBError'
    this.code = code
    this.retryable = retryable
  }
}
```

/**
 * Type guard to check if an error is an ArcherDB error.
 *
 * @example
 * ```typescript
 * try {
 *   await client.queryRadius(filter)
 * } catch (error) {
 *   if (isArcherDBError(error)) {
 *     console.log(`ArcherDB error: ${error.code}`)
 *   }
 * }
 * ```

export function isArcherDBError(error: unknown): error is ArcherDBError {
  return error instanceof ArcherDBError
}

/** Check if error is a network error */
export function isNetworkError(error: unknown): error is ConnectionError | ClusterError {
  return error instanceof ConnectionError || error instanceof ClusterError
}

/** Check if error is a validation error */
export function isValidationError(error: unknown): error is ValidationError {
  return error instanceof ValidationError
}

/** Check if error is retryable */
export function isRetryable(error: unknown): boolean {
  return isArcherDBError(error) && error.retryable
}

## Error Handling

ArcherDB uses typed errors with numeric codes:

```typescript
import {
  ArcherDBError,
  ValidationError,
  ConnectionError,
  isArcherDBError,
  isNetworkError,
  isRetryable,
} from 'archerdb-node'

try {
  const results = await client.queryRadius(options)
} catch (error) {
  if (error instanceof ValidationError) {
    // Invalid input (e.g., coordinates out of range)
    console.error(`Validation error (${error.code}): ${error.message}`)
    return
  }

  if (isNetworkError(error)) {
    // Network issues - may be retryable
    console.error(`Network error: ${error.message}`)
    if (isRetryable(error)) {
      // Implement retry logic
    }
    return
  }

  if (isArcherDBError(error)) {
    // Other ArcherDB errors
    console.error(`ArcherDB error (${error.code}): ${error.message}`)
    return
  }

  // Unknown error
  throw error
}
```
