// IsNetworkError returns true if err is any network-related error.
func IsNetworkError(err error) bool {
    return errors.Is(err, ErrConnectionFailed) ||
           errors.Is(err, ErrConnectionTimeout) ||
           errors.Is(err, ErrClusterUnavailable)
}

// IsValidationError returns true if err is any validation error.
func IsValidationError(err error) bool {
    return errors.Is(err, ErrInvalidCoordinates) ||
           errors.Is(err, ErrBatchTooLarge) ||

}

## Error Handling

ArcherDB errors support the standard `errors.Is` pattern:

```go
import "github.com/archerdb/archerdb-go/pkg/errors"

result, err := client.QueryRadius(filter)
if err != nil {
    // Check specific error type
    if errors.Is(err, archerdb.ErrInvalidCoordinates) {
        log.Printf("Invalid coordinates: %v", err)
        return
    }

    // Check error category
    if errors.IsNetworkError(err) {
        log.Printf("Network error, will retry: %v", err)
        // Retry logic
    }

    // Check if retryable
    var geoErr archerdb.GeoError
    if errors.As(err, &geoErr) && geoErr.Retryable() {
        // Implement retry
    }
}
```

### Error Categories

| Category | Error Types | Action |
|----------|-------------|--------|
| Network | ConnectionFailed, ConnectionTimeout, ClusterUnavailable | Retry with backoff |
| Validation | InvalidCoordinates, BatchTooLarge, InvalidEntityID | Fix input |
| State | EntityExpired, NotPrimary | Depends on error |
