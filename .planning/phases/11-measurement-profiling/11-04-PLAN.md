     pub const StatisticalResult = struct {
         mean_ns: f64,
         std_dev_ns: f64,
         confidence_interval_95: struct { lower_ns: f64, upper_ns: f64 },
         min_ns: f64,
         max_ns: f64,
         p50_ns: f64,
         p99_ns: f64,
         samples: usize,
         outliers_removed: usize,

         /// Format as human-readable output
         pub fn format(self: *const @This(), writer: anytype) !void {
             try writer.print(
                 \\mean={d:.3}ms (+/- {d:.3}ms)
                 \\95% CI: [{d:.3}ms, {d:.3}ms]
                 \\P50={d:.3}ms P99={d:.3}ms min={d:.3}ms max={d:.3}ms
                 \\samples={d} (outliers removed: {d})
             , .{
                 self.mean_ns / 1e6, self.std_dev_ns / 1e6,
                 self.confidence_interval_95.lower_ns / 1e6,
                 self.confidence_interval_95.upper_ns / 1e6,
                 self.p50_ns / 1e6, self.p99_ns / 1e6,
                 self.min_ns / 1e6, self.max_ns / 1e6,
                 self.samples, self.outliers_removed,
             });
         }

         /// Compare with baseline, return true if regression detected
         pub fn isRegression(self: *const @This(), baseline: *const @This()) bool {
             // Regression if current mean > baseline mean + 2 * baseline stddev
             const threshold = baseline.mean_ns + 2 * baseline.std_dev_ns;
             return self.mean_ns > threshold;
         }

         /// Format comparison with baseline
         pub fn formatComparison(self: *const @This(), baseline: *const @This(), writer: anytype) !void {
             const delta_percent = ((self.mean_ns - baseline.mean_ns) / baseline.mean_ns) * 100;
             const verdict = if (self.isRegression(baseline))
                 "REGRESSION"
             else if (delta_percent < -5)
                 "IMPROVEMENT"
             else
                 "NO CHANGE";
             try writer.print("{s}: {d:.1}% ({d:.3}ms -> {d:.3}ms)\n", .{
                 verdict, delta_percent, baseline.mean_ns / 1e6, self.mean_ns / 1e6,
             });
         }
     };
     ```

  2. Add computeStatistics function:
     ```zig
     /// Compute statistical analysis from duration samples
     pub fn computeStatistics(durations: []Duration) StatisticalResult {
         // Convert to nanoseconds array
         var samples: [256]f64 = undefined;
         const n = @min(durations.len, 256);
         for (durations[0..n], 0..) |d, i| {
             samples[i] = @floatFromInt(d.total_ns());
         }

         // Sort for percentiles
         std.sort.block(f64, samples[0..n], {}, std.sort.asc(f64));

         // Remove outliers using IQR method
         const q1_idx = n / 4;
         const q3_idx = (n * 3) / 4;
         const iqr = samples[q3_idx] - samples[q1_idx];
         const lower_bound = samples[q1_idx] - 1.5 * iqr;
         const upper_bound = samples[q3_idx] + 1.5 * iqr;

         var filtered: [256]f64 = undefined;
         var filtered_count: usize = 0;
         var outliers: usize = 0;
         for (samples[0..n]) |s| {
             if (s >= lower_bound and s <= upper_bound) {
                 filtered[filtered_count] = s;
                 filtered_count += 1;
             } else {
                 outliers += 1;
             }
         }

         // Compute mean and stddev
         var sum: f64 = 0;
         for (filtered[0..filtered_count]) |s| sum += s;
         const mean = sum / @as(f64, @floatFromInt(filtered_count));

         var variance_sum: f64 = 0;
         for (filtered[0..filtered_count]) |s| {
             const diff = s - mean;
             variance_sum += diff * diff;
         }
         const std_dev = @sqrt(variance_sum / @as(f64, @floatFromInt(filtered_count)));
         const se = std_dev / @sqrt(@as(f64, @floatFromInt(filtered_count)));

         return .{
             .mean_ns = mean,
             .std_dev_ns = std_dev,
             .confidence_interval_95 = .{
                 .lower_ns = mean - 1.96 * se,
                 .upper_ns = mean + 1.96 * se,
             },
             .min_ns = filtered[0],
             .max_ns = filtered[filtered_count - 1],
             .p50_ns = filtered[filtered_count / 2],
             .p99_ns = filtered[(filtered_count * 99) / 100],
             .samples = filtered_count,
             .outliers_removed = outliers,
         };
     }
     ```

  3. Add runWithStatistics method to Bench:
     ```zig
     /// Run benchmark multiple times and return statistical result
     pub fn runWithStatistics(
         bench: *Bench,
         comptime runs: usize,
         comptime func: fn (*Bench) Duration,
     ) StatisticalResult {
         var durations: [runs]Duration = undefined;
         for (&durations) |*d| {
             d.* = func(bench);
         }
         return computeStatistics(&durations);
     }
     ```

  <name>Task 3: Create GitHub Actions benchmark workflow</name>
  <files>.github/workflows/benchmark.yml</files>
  <action>
  Create .github/workflows/benchmark.yml for CI benchmark integration:

  ```yaml
  name: Performance Benchmarks

  on:
    push:
      branches: [main]
    pull_request:
      branches: [main]

  jobs:
    benchmark:
      runs-on: ubuntu-latest
      permissions:
        contents: read
        pull-requests: write

      steps:
        - uses: actions/checkout@v4
          with:
            fetch-depth: 0  # Need history for baseline comparison

        - name: Install dependencies
          run: |
            sudo apt-get update
            sudo apt-get install -y linux-perf jq

        - name: Download baseline (main branch)
          if: github.event_name == 'pull_request'
          uses: actions/download-artifact@v4
          with:
            name: benchmark-baseline
            path: baseline/
          continue-on-error: true  # OK if no baseline exists yet

        - name: Run benchmarks
          run: |
            MODE="quick"
            if [[ "${{ github.event_name }}" == "push" && "${{ github.ref }}" == "refs/heads/main" ]]; then
              MODE="full"
            fi
            ./scripts/benchmark-ci.sh --mode "$MODE" --output benchmark-results.json

        - name: Compare with baseline
          if: github.event_name == 'pull_request' && hashFiles('baseline/benchmark-results.json') != ''
          run: |
            ./scripts/benchmark-ci.sh --compare \
              --baseline baseline/benchmark-results.json \
              --output benchmark-results.json \
              > comparison.txt 2>&1 || REGRESSION=1

            # Post comment with results
            if [[ -f comparison.txt ]]; then
              echo "## Performance Benchmark Results" > comment.md
              echo "" >> comment.md
              echo '```' >> comment.md
              cat comparison.txt >> comment.md
              echo '```' >> comment.md

              if [[ "${REGRESSION:-0}" == "1" ]]; then
                echo "" >> comment.md
                echo "**Warning: Performance regression detected!**" >> comment.md
              fi
            fi

        - name: Upload results as baseline
          if: github.event_name == 'push' && github.ref == 'refs/heads/main'
          uses: actions/upload-artifact@v4
          with:
            name: benchmark-baseline
            path: benchmark-results.json
            retention-days: 90

        - name: Post PR comment
          if: github.event_name == 'pull_request' && hashFiles('comment.md') != ''
          uses: actions/github-script@v7
          with:
            script: |
              const fs = require('fs');
              const comment = fs.readFileSync('comment.md', 'utf8');
              github.rest.issues.createComment({
                issue_number: context.issue.number,
                owner: context.repo.owner,
                repo: context.repo.repo,
                body: comment
              });
  ```

  Key features:
  - Quick benchmarks on PRs, full benchmarks on main merge
  - Baseline stored as artifact (90 day retention)
  - PR comment with comparison results
  - Warning on performance regression
  </action>
