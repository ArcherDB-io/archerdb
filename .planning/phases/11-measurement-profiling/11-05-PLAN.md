  ```zig
  // SPDX-License-Identifier: Apache-2.0
  // Copyright (c) 2024-2025 ArcherDB Contributors
  //! Tracy profiler zone helpers.
  //!
  //! Usage:
  //!   const tracy = @import("testing/tracy_zones.zig");
  //!
  //!   fn myFunction() void {
  //!       const zone = tracy.zone(@src(), "my_function");
  //!       defer zone.end();
  //!       // ... function body ...
  //!   }
  //!
  //! When Tracy is disabled (default builds), these are no-ops with zero overhead.

  const std = @import("std");

  // Conditional import - ztracy only available in profile builds
  const ztracy = if (@hasDecl(@import("root"), "ztracy"))
      @import("ztracy")
  else
      null;

  pub const Zone = if (ztracy) |t| t.ZoneCtx else struct {
      pub fn end(_: @This()) void {}
      pub fn text(_: @This(), _: []const u8) void {}
      pub fn name(_: @This(), _: []const u8) void {}
      pub fn value(_: @This(), _: u64) void {}
  };

  /// Create a named zone for profiling
  pub inline fn zone(src: std.builtin.SourceLocation, comptime zone_name: []const u8) Zone {
      if (ztracy) |t| {
          return t.zone(src, zone_name);
      }
      return .{};
  }

  /// Create a zone with runtime name
  pub inline fn zoneN(src: std.builtin.SourceLocation, name_str: []const u8) Zone {
      if (ztracy) |t| {
          var z = t.zone(src, null);
          z.name(name_str);
          return z;
      }
      return .{};
  }

  /// Frame marker for main loop iterations
  pub inline fn frameMark() void {
      if (ztracy) |t| {
          t.frameMark();
      }
  }

  /// Named frame marker (for multiple frame types)
  pub inline fn frameMarkNamed(comptime name: []const u8) void {
      if (ztracy) |t| {
          t.frameMarkNamed(name);
      }
  }

  /// Log a message to Tracy timeline
  pub inline fn message(text: []const u8) void {
      if (ztracy) |t| {
          t.message(text);
      }
  }

  /// Log a message with color
  pub inline fn messageColor(text: []const u8, color: u32) void {
      if (ztracy) |t| {
          t.messageColor(text, color);
      }
  }

  // Pre-defined zone colors for consistency
  pub const colors = struct {
      pub const query: u32 = 0x00FF00;      // Green
      pub const storage: u32 = 0x0000FF;    // Blue
      pub const consensus: u32 = 0xFF0000;  // Red
      pub const network: u32 = 0xFFFF00;    // Yellow
      pub const index: u32 = 0xFF00FF;      // Magenta
  };
  ```

  ```bash
  #!/usr/bin/env bash
  # Deploy Parca agent for continuous profiling
  #
  # Parca provides always-on profiling with <1% overhead via eBPF.
  # Profiles are stored and can be analyzed via the Parca web UI.
  #
  # Usage:
  #   ./scripts/parca-agent.sh start [--server <url>]
  #   ./scripts/parca-agent.sh stop
  #   ./scripts/parca-agent.sh status

  set -euo pipefail

  PARCA_AGENT_VERSION="${PARCA_AGENT_VERSION:-0.33.0}"
  PARCA_SERVER="${PARCA_SERVER:-http://localhost:7070}"
  PARCA_AGENT_BIN="${PARCA_AGENT_BIN:-/usr/local/bin/parca-agent}"

  usage() {
      cat <<EOF
  Usage: $0 <command> [options]

  Commands:
      start           Start Parca agent
      stop            Stop Parca agent
      status          Check agent status
      install         Download and install Parca agent

  Options:
      --server <url>  Parca server URL (default: http://localhost:7070)
      --version <v>   Parca agent version (default: $PARCA_AGENT_VERSION)

  Prerequisites:
      - Linux kernel >= 5.6 with eBPF support
      - Root privileges for eBPF programs
      - Parca server running (or use Parca Cloud)

  Quick Start:
      # Install agent
      sudo ./scripts/parca-agent.sh install

      # Start with local Parca server
      sudo ./scripts/parca-agent.sh start

      # Or connect to Parca Cloud
      sudo ./scripts/parca-agent.sh start --server https://grpc.polarsignals.com:443
  EOF
  }

  install_agent() {
      echo "Installing Parca agent v${PARCA_AGENT_VERSION}..."
      curl -sL "https://github.com/parca-dev/parca-agent/releases/download/v${PARCA_AGENT_VERSION}/parca-agent_${PARCA_AGENT_VERSION}_linux_amd64.tar.gz" | \
          sudo tar -xzf - -C /usr/local/bin parca-agent
      echo "Installed to ${PARCA_AGENT_BIN}"
  }

  start_agent() {
      echo "Starting Parca agent..."
      sudo "${PARCA_AGENT_BIN}" \
          --remote-store-address="${PARCA_SERVER}" \
          --node="$(hostname)" \
          --log-level=info &
      echo "Parca agent started (PID: $!)"
  }

  stop_agent() {
      echo "Stopping Parca agent..."
      sudo pkill -f parca-agent || echo "Agent not running"
  }

  status_agent() {
      if pgrep -f parca-agent > /dev/null; then
          echo "Parca agent is running"
          pgrep -af parca-agent
      else
          echo "Parca agent is not running"
      fi
  }

  # Parse arguments
  COMMAND="${1:-}"
  shift || true

  while [[ $# -gt 0 ]]; do
      case $1 in
          --server) PARCA_SERVER="$2"; shift 2 ;;
          --version) PARCA_AGENT_VERSION="$2"; shift 2 ;;
          --help|-h) usage; exit 0 ;;
          *) echo "Unknown option: $1"; usage; exit 1 ;;
      esac
  done

  case "${COMMAND}" in
      start) start_agent ;;
      stop) stop_agent ;;
      status) status_agent ;;
      install) install_agent ;;
      *) usage; exit 1 ;;
  esac
  ```
