fn run_warmup(b: *GeoBenchmark) void {
    const target_variance = 0.05; // 5% variance threshold
    var samples: [30]u64 = undefined;
    var sample_idx: usize = 0;

    while (true) {
        // Run sample batch
        const latency = b.run_single_batch();
        samples[sample_idx % 30] = latency;
        sample_idx += 1;

        if (sample_idx >= 30) {
            const variance = calculate_variance(&samples);
            if (variance < target_variance) {
                log.info("Warmup complete after {} iterations (variance: {d:.2}%)", .{
                    sample_idx, variance * 100,
                });
                break;
            }
        }
    }
}

```bash
#!/usr/bin/env bash
#
# run-benchmarks.sh - Run full ArcherDB benchmark suite
#
# Usage: ./scripts/run-benchmarks.sh [--quick|--full|--competitor]
#
# This script runs benchmarks per CONTEXT.md methodology:
# - Multiple concurrency levels (1, 10, 100)
# - Batch size variations (1, 100, 1000, 10000)
# - Cold and warm cache
# - With and without encryption
# - Single node and 3-node cluster
# - 30+ runs for statistical significance

set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
ROOT_DIR="$(dirname "$SCRIPT_DIR")"
RESULTS_DIR="${ROOT_DIR}/benchmark-results"
TIMESTAMP=$(date +%Y%m%d-%H%M%S)

# Configuration
MODE="${1:---quick}"
ENTITY_COUNTS=(100000)  # Default: 100K
EVENT_MULTIPLIER=10      # Events = entities * multiplier
CONCURRENCY_LEVELS=(1 10 100)
BATCH_SIZES=(1 100 1000 10000)
RUNS=30

case "$MODE" in
  --quick)
    ENTITY_COUNTS=(10000)
    RUNS=5
    ;;
  --full)
    ENTITY_COUNTS=(100000 1000000 10000000)
    RUNS=30
    ;;
  --extreme)
    ENTITY_COUNTS=(10000000 100000000)
    RUNS=30
    ;;
esac

mkdir -p "$RESULTS_DIR"

echo "=== ArcherDB Benchmark Suite ==="
echo "Mode: $MODE"
echo "Runs per config: $RUNS"
echo "Results: $RESULTS_DIR"
echo ""

# Build release binary
echo "Building ArcherDB (release)..."
cd "$ROOT_DIR"
./zig/zig build -Drelease

ARCHERDB="$ROOT_DIR/zig-out/bin/archerdb"

# Function to run benchmark and collect results
run_benchmark() {
    local entities=$1
    local concurrency=$2
    local batch_size=$3
    local encrypted=$4
    local run_num=$5

    local events=$((entities * EVENT_MULTIPLIER))
    local config_name="e${entities}_c${concurrency}_b${batch_size}"
    [ "$encrypted" = "true" ] && config_name="${config_name}_enc"

    local output_file="$RESULTS_DIR/${config_name}_run${run_num}.json"

    echo "  Run $run_num: entities=$entities, concurrency=$concurrency, batch=$batch_size, encrypted=$encrypted"

    $ARCHERDB benchmark \
        --event-count="$events" \
        --entity-count="$entities" \
        --clients="$concurrency" \
        --query-uuid-count=1000 \
        --query-radius-count=100 \
        --query-polygon-count=100 \
        > "$output_file" 2>&1 || true
}

# Run benchmark matrix
for entities in "${ENTITY_COUNTS[@]}"; do
    echo ""
    echo "=== Entity count: $entities ==="

    for concurrency in "${CONCURRENCY_LEVELS[@]}"; do
        echo "Concurrency: $concurrency"

        for run in $(seq 1 $RUNS); do
            run_benchmark "$entities" "$concurrency" 1000 "false" "$run"
        done
    done
done

# Aggregate results
echo ""
echo "=== Aggregating results ==="
python3 "$SCRIPT_DIR/aggregate-benchmarks.py" "$RESULTS_DIR" > "$RESULTS_DIR/summary-$TIMESTAMP.json"

echo ""
echo "Benchmark complete. Results in: $RESULTS_DIR/summary-$TIMESTAMP.json"
```

Create scripts/aggregate-benchmarks.py for result aggregation:
```python
#!/usr/bin/env python3
"""Aggregate benchmark results into summary statistics."""
import json
import os
import re
import statistics
import sys
from pathlib import Path

def parse_result_file(filepath):
    """Parse a single benchmark output file."""
    results = {}
    with open(filepath) as f:
        content = f.read()

    # Parse throughput
    match = re.search(r'throughput = (\d+) events/s', content)
    if match:
        results['insert_throughput'] = int(match.group(1))

    # Parse latencies
    for metric, pattern in [
        ('insert_p99', r'insert batch latency p99\s*=\s*(\d+)\s*ms'),
        ('uuid_p99', r'UUID query latency p99\s*=\s*(\d+)\s*ms'),
        ('radius_p99', r'radius query latency p99\s*=\s*(\d+)\s*ms'),
        ('polygon_p99', r'polygon query latency p99\s*=\s*(\d+)\s*ms'),
    ]:
        match = re.search(pattern, content)
        if match:
            results[metric] = int(match.group(1))

    return results

def aggregate_results(results_dir):
    """Aggregate all result files into summary statistics."""
    configs = {}

    for filepath in Path(results_dir).glob('*.json'):
        if 'summary' in filepath.name:
            continue

        # Parse config from filename
        match = re.match(r'e(\d+)_c(\d+)_b(\d+)(_enc)?_run(\d+)', filepath.stem)
        if not match:
            continue

        config = f"e{match.group(1)}_c{match.group(2)}_b{match.group(3)}"
        if match.group(4):
            config += "_enc"

        if config not in configs:
            configs[config] = []

        results = parse_result_file(filepath)
        if results:
            configs[config].append(results)

    # Calculate statistics for each config
    summary = {}
    for config, runs in configs.items():
        if not runs:
            continue

        summary[config] = {}
        for metric in runs[0].keys():
            values = [r[metric] for r in runs if metric in r]
            if values:
                summary[config][metric] = {
                    'mean': statistics.mean(values),
                    'median': statistics.median(values),
                    'stdev': statistics.stdev(values) if len(values) > 1 else 0,
                    'min': min(values),
                    'max': max(values),
                    'runs': len(values),
                }

    print(json.dumps(summary, indent=2))

if __name__ == '__main__':
    aggregate_results(sys.argv[1])
```

# ArcherDB Hardware Requirements

## Overview

ArcherDB's performance scales with hardware resources. This guide provides
minimum and recommended specifications for different deployment scales.

## Memory Sizing Formula (PERF-08, PERF-09)

**Formula:** `RAM (bytes) = entity_count * 64 / 0.70`

The divisor (0.70) accounts for memory overhead and working set.

| Entities    | Minimum RAM | Recommended RAM |
|-------------|-------------|-----------------|
| 1 million   | 128 MB      | 256 MB          |
| 10 million  | 1.3 GB      | 2.5 GB          |
| 100 million | 13 GB       | 25 GB           |
| 1 billion   | 130 GB      | 256 GB          |

## Deployment Profiles

### Development / Testing
- **CPU:** 2 cores
- **RAM:** 4 GB
- **Storage:** 20 GB SSD
- **Capacity:** ~10 million entities

### Small Production (< 100M entities)
- **CPU:** 4 cores (modern x86_64 or ARM64 with AES-NI)
- **RAM:** 16 GB
- **Storage:** 100 GB NVMe SSD
- **Network:** 1 Gbps

**Cloud equivalents:**
- AWS: m6i.xlarge
- GCP: n2-standard-4
- Azure: Standard_D4s_v5

### Medium Production (100M - 1B entities)
- **CPU:** 8-16 cores
- **RAM:** 64-128 GB
- **Storage:** 500 GB - 1 TB NVMe SSD
- **Network:** 10 Gbps

**Cloud equivalents:**
- AWS: r6i.4xlarge
- GCP: n2-highmem-16
- Azure: Standard_E16s_v5

### Large Production (> 1B entities)
- **CPU:** 32+ cores
- **RAM:** 256+ GB
- **Storage:** 2+ TB NVMe SSD (consider RAID)
- **Network:** 25 Gbps

**Cloud equivalents:**
- AWS: r6i.16xlarge or x2idn.xlarge
- GCP: n2-highmem-64
- Azure: Standard_E64s_v5

## Storage Recommendations

- **Type:** NVMe SSD required for production workloads
- **Capacity:** 3-5x data size for compaction headroom
- **IOPS:** 100K+ random read IOPS recommended
- **Throughput:** 3+ GB/s sequential write

SATA SSDs work for development but will limit write throughput.

## Network Requirements

- **Latency:** < 1ms between replicas (same region)
- **Bandwidth:** 1 Gbps minimum, 10 Gbps recommended for replication
- **Ports:** 3000 (default), configurable

## Workload-Specific Tuning

### Read-Heavy (> 90% queries)
- Maximize RAM for entity index cache
- SSD IOPS more important than sequential throughput
- Consider read replicas

### Write-Heavy (> 50% inserts)
- Fast sequential write throughput critical
- Ensure adequate compaction headroom (4x data size)
- Monitor write amplification metrics

### Balanced
- Start with recommended specs
- Monitor metrics and scale as needed
- Consider geographic distribution for latency

## Multi-Region Deployment

For multi-region deployments, see [Replication Documentation](operations-runbook.md#replication).

Key considerations:
- S3-compatible storage for async replication
- Higher network latency tolerance (100-200ms)
- Regional read replicas

## Kubernetes Deployment

For Kubernetes resource requests/limits, see [Operations Runbook](operations-runbook.md#kubernetes).
