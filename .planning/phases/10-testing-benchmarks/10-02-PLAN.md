```zig
//! Integration tests for backup/restore functionality (INT-03)
//!
//! Tests full backup/restore cycle:
//! 1. Create data
//! 2. Take backup
//! 3. Corrupt/delete data
//! 4. Restore from backup
//! 5. Verify data integrity

const std = @import("std");
const TmpArcherDB = @import("tmp_archerdb.zig");

test "integration: backup and restore cycle" {
    const allocator = std.testing.allocator;
    var archerdb = try TmpArcherDB.init(allocator, .{});
    defer archerdb.deinit(allocator);

    // 1. Insert test data
    const test_entity_id: u128 = 0x12345678;
    try archerdb.insertEvent(createTestEvent(test_entity_id));

    // 2. Create backup
    const backup_path = try archerdb.createBackup();
    defer std.fs.deleteFileAbsolute(backup_path) catch {};

    // 3. Verify backup file exists and has content
    const backup_stat = try std.fs.cwd().statFile(backup_path);
    try std.testing.expect(backup_stat.size > 0);

    // 4. Corrupt the original data (simulate disaster)
    try archerdb.stop();
    try std.fs.deleteFileAbsolute(archerdb.data_file);

    // 5. Restore from backup
    try archerdb.restore(backup_path);
    try archerdb.start();

    // 6. Verify data is intact
    const result = try archerdb.queryUuid(test_entity_id);
    try std.testing.expect(result != null);
}

test "integration: backup during active writes" {
    // Test that backup works while writes are happening
    // Verify consistency of backup snapshot
}
```

```zig
//! Integration tests for encryption at rest (INT-06)
//!
//! Tests:
//! - Data encrypted on disk (not readable as plaintext)
//! - Correct decryption with valid key
//! - Decryption fails with wrong key
//! - Key rotation works correctly

const std = @import("std");
const TmpArcherDB = @import("tmp_archerdb.zig");

test "integration: encryption at rest" {
    const allocator = std.testing.allocator;

    // Generate test encryption key
    var key: [32]u8 = undefined;
    std.crypto.random.bytes(&key);

    // Start with encryption enabled
    var archerdb = try TmpArcherDB.init(allocator, .{
        .encryption_key = &key,
    });
    defer archerdb.deinit(allocator);

    // Insert recognizable data
    const test_content: u128 = 0xDEADBEEFCAFEBABE;
    try archerdb.insertEvent(createTestEventWithContent(test_content));

    // Stop server and examine data file
    try archerdb.stop();

    // Verify data file doesn't contain plaintext content
    const data = try std.fs.cwd().readFileAlloc(allocator, archerdb.data_file, 10 * 1024 * 1024);
    defer allocator.free(data);

    // The test content should NOT appear in plaintext
    const test_bytes = std.mem.asBytes(&test_content);
    try std.testing.expect(std.mem.indexOf(u8, data, test_bytes) == null);

    // Restart with correct key - data should be accessible
    try archerdb.start();
    const result = try archerdb.queryByContent(test_content);
    try std.testing.expect(result != null);
}
```
