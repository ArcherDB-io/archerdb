```bash
-- Create table matching ArcherDB schema
CREATE TABLE geo_events (

    entity_id UUID NOT NULL,
    location GEOGRAPHY(POINT, 4326) NOT NULL,
    timestamp TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    velocity_mps FLOAT,
```

```python
#!/usr/bin/env python3
"""Common utilities for competitor benchmarks."""
import random
import statistics
import time
from dataclasses import dataclass
from typing import List

@dataclass
class BenchmarkResult:
    """Results from a benchmark run."""
    operation: str
    system: str
    total_ops: int
    duration_ms: float
    ops_per_sec: float
    latency_p50_us: float
    latency_p95_us: float
    latency_p99_us: float
    latency_p999_us: float
    errors: int

def percentile(data: List[float], p: float) -> float:
    """Calculate percentile."""
    if not data:
        return 0.0
    sorted_data = sorted(data)
    k = (len(sorted_data) - 1) * p / 100
    f = int(k)
    c = f + 1 if f + 1 < len(sorted_data) else f
    return sorted_data[f] + (k - f) * (sorted_data[c] - sorted_data[f])

def generate_random_event():
    """Generate a random geo event matching ArcherDB workload."""
    return {
        'entity_id': random.getrandbits(128),
        'lat': random.uniform(-90, 90),
        'lon': random.uniform(-180, 180),
        'velocity_mps': random.uniform(0, 50),
        'heading': random.uniform(0, 360),
    }

def run_with_timing(operation, warmup_count=100, run_count=1000):
    """Run operation with timing collection."""
    # Warmup
    for _ in range(warmup_count):
        operation()

    # Timed runs
    latencies = []
    start = time.perf_counter()
    for _ in range(run_count):
        op_start = time.perf_counter()
        operation()
        latencies.append((time.perf_counter() - op_start) * 1_000_000)  # us
    duration_ms = (time.perf_counter() - start) * 1000

    return {
        'count': run_count,
        'duration_ms': duration_ms,
        'ops_per_sec': run_count / (duration_ms / 1000),
        'p50': percentile(latencies, 50),
        'p95': percentile(latencies, 95),
        'p99': percentile(latencies, 99),
        'p999': percentile(latencies, 99.9),
    }
```

```python
#!/usr/bin/env python3
"""PostGIS benchmark driver (BENCH-03)."""
import json
import psycopg2
from psycopg2.extras import execute_batch
from common import generate_random_event, run_with_timing, percentile
import time
import random
import sys

def main(entity_count=100000, query_count=1000):
    conn = psycopg2.connect(
        host='localhost', port=5432,
        user='benchmark', password='benchmark', dbname='benchmark'
    )
    cur = conn.cursor()

    results = []
    entity_ids = []

    # INSERT benchmark
    print(f"PostGIS: Inserting {entity_count} events...")
    latencies = []
    batch_size = 1000
    start = time.perf_counter()

    for i in range(0, entity_count, batch_size):
        batch = []
        for _ in range(min(batch_size, entity_count - i)):
            e = generate_random_event()
            entity_ids.append(str(e['entity_id']))
            batch.append((
                str(e['entity_id']),
                f"POINT({e['lon']} {e['lat']})",
                e['velocity_mps'],
                e['heading']
            ))

        batch_start = time.perf_counter()
        execute_batch(cur,
            "INSERT INTO geo_events (entity_id, location, velocity_mps, heading) "
            "VALUES (%s, ST_GeogFromText(%s), %s, %s)",
            batch
        )
        conn.commit()
        latencies.append((time.perf_counter() - batch_start) * 1_000_000)

    duration_ms = (time.perf_counter() - start) * 1000
    results.append({
        'operation': 'INSERT',
        'system': 'PostGIS',
        'total_ops': entity_count,
        'duration_ms': duration_ms,
        'ops_per_sec': entity_count / (duration_ms / 1000),
        'p99_ms': percentile(latencies, 99) / 1000,
    })

    # RADIUS QUERY benchmark
    print(f"PostGIS: Running {query_count} radius queries...")
    latencies = []
    start = time.perf_counter()

    for _ in range(query_count):
        lat, lon = random.uniform(-90, 90), random.uniform(-180, 180)
        q_start = time.perf_counter()
        cur.execute("""
            SELECT entity_id, ST_X(location::geometry), ST_Y(location::geometry)
            FROM geo_events
            WHERE ST_DWithin(location, ST_GeogFromText(%s), %s)
            LIMIT 100
        """, (f"POINT({lon} {lat})", 10000))  # 10km radius
        cur.fetchall()
        latencies.append((time.perf_counter() - q_start) * 1_000_000)

    duration_ms = (time.perf_counter() - start) * 1000
    results.append({
        'operation': 'RADIUS_QUERY',
        'system': 'PostGIS',
        'total_ops': query_count,
        'duration_ms': duration_ms,
        'ops_per_sec': query_count / (duration_ms / 1000),
        'p50_ms': percentile(latencies, 50) / 1000,
        'p95_ms': percentile(latencies, 95) / 1000,
        'p99_ms': percentile(latencies, 99) / 1000,
        'p999_ms': percentile(latencies, 99.9) / 1000,
    })

    # UUID QUERY benchmark
    print(f"PostGIS: Running {query_count} UUID queries...")
    latencies = []
    start = time.perf_counter()

    for _ in range(query_count):
        entity_id = random.choice(entity_ids)
        q_start = time.perf_counter()
        cur.execute(
            "SELECT * FROM geo_events WHERE entity_id = %s ORDER BY timestamp DESC LIMIT 1",
            (entity_id,)
        )
        cur.fetchone()
        latencies.append((time.perf_counter() - q_start) * 1_000_000)

    duration_ms = (time.perf_counter() - start) * 1000
    results.append({
        'operation': 'UUID_QUERY',
        'system': 'PostGIS',
        'total_ops': query_count,
        'duration_ms': duration_ms,
        'ops_per_sec': query_count / (duration_ms / 1000),
        'p50_ms': percentile(latencies, 50) / 1000,
        'p95_ms': percentile(latencies, 95) / 1000,
        'p99_ms': percentile(latencies, 99) / 1000,
        'p999_ms': percentile(latencies, 99.9) / 1000,
    })

    cur.close()
    conn.close()

    print(json.dumps(results, indent=2))
    return results

if __name__ == '__main__':
    entity_count = int(sys.argv[1]) if len(sys.argv) > 1 else 100000
    main(entity_count)
```

```bash
#
# run-comparison.sh - Run ArcherDB vs competitor benchmarks
```

```python
"""Generate comparison report from benchmark results."""
```
