Create dedicated VOPR workflow for long-running fuzzing:

1. Create .github/workflows/vopr.yml:
```yaml
name: VOPR Fuzzer
on:
  schedule:
    - cron: '0 2 * * *'  # Daily at 2 AM UTC
  workflow_dispatch:  # Allow manual trigger

jobs:
  vopr-geo:
    runs-on: ubuntu-latest
    timeout-minutes: 180  # 3 hours max (2h run + margin)
    steps:
      - uses: actions/checkout@v4
      - name: Download Zig
        run: ./zig/download.sh
      - name: Run VOPR (geo state machine)
        run: ./.github/ci/run-vopr.sh geo 7200  # 2 hours
      - name: Upload VOPR logs
        if: failure()
        uses: actions/upload-artifact@v4
        with:
          name: vopr-geo-logs
          path: vopr-*.log
          retention-days: 30

  vopr-testing:
    runs-on: ubuntu-latest
    timeout-minutes: 180
    steps:
      - uses: actions/checkout@v4
      - name: Download Zig
        run: ./zig/download.sh
      - name: Run VOPR (testing state machine)
        run: ./.github/ci/run-vopr.sh testing 7200
      - name: Upload VOPR logs
        if: failure()
        uses: actions/upload-artifact@v4
        with:
          name: vopr-testing-logs
          path: vopr-*.log
          retention-days: 30
```

2. Create .github/ci/run-vopr.sh:
```bash
#!/usr/bin/env bash
set -euo pipefail

STATE_MACHINE="${1:-geo}"
DURATION_SECS="${2:-7200}"  # 2 hours default

echo "Running VOPR with state machine: $STATE_MACHINE for $DURATION_SECS seconds"

# Build VOPR
./zig/zig build vopr \
  -Dvopr-state-machine="$STATE_MACHINE" \
  -Drelease

# Run with timeout
timeout --signal=SIGINT "$DURATION_SECS" \
  ./zig-out/bin/vopr "${GITHUB_SHA:-$(git rev-parse HEAD)}" \
  --ticks-max-requests=100000000 \
  2>&1 | tee "vopr-${STATE_MACHINE}-$(date +%Y%m%d-%H%M%S).log" || {
    EXIT_CODE=$?
    if [ $EXIT_CODE -eq 124 ]; then
      echo "VOPR completed (timeout reached - this is expected)"
      exit 0
    fi
    echo "VOPR failed with exit code $EXIT_CODE"
    exit $EXIT_CODE
}
```

```yaml
benchmark:
  needs: test
  if: github.ref == 'refs/heads/main'
  runs-on: ubuntu-latest
  steps:
    - uses: actions/checkout@v4
    - name: Download Zig
      run: ./zig/download.sh

    - name: Run benchmarks
      run: |
        ./zig/zig build -Drelease
        ./zig-out/bin/archerdb benchmark \
          --event-count=10000 \
          --entity-count=1000 \
          --query-uuid-count=1000 \
          --query-radius-count=100 \
          --query-polygon-count=100 \
          > benchmark-results.txt 2>&1
        # Parse results into JSON format for github-action-benchmark
        python3 .github/ci/parse-benchmark.py benchmark-results.txt > benchmark-results.json

    - name: Store benchmark result
      uses: benchmark-action/github-action-benchmark@v1
      with:
        tool: 'customSmallerIsBetter'
        output-file-path: benchmark-results.json
        github-token: ${{ secrets.GITHUB_TOKEN }}
        auto-push: true
        alert-threshold: '105%'  # 5% regression
        comment-on-alert: true
        fail-on-alert: true
        benchmark-data-dir-path: 'benchmark-data'
```

2. Create .github/ci/parse-benchmark.py:
```python
#!/usr/bin/env python3
"""Parse ArcherDB benchmark output into github-action-benchmark JSON format."""
import json
import re
import sys

def parse_benchmark(filename):
    results = []
    with open(filename) as f:
        content = f.read()

    # Parse insert throughput
    match = re.search(r'throughput = (\d+) events/s', content)
    if match:
        results.append({
            'name': 'Insert Throughput',
            'unit': 'events/s',
            'value': int(match.group(1)),
            'biggerIsBetter': True
        })

    # Parse latency percentiles
    for op, label in [('insert batch', 'Insert'), ('UUID query', 'UUID'),
                      ('radius query', 'Radius'), ('polygon query', 'Polygon')]:
        match = re.search(rf'{op} latency p99\s*=\s*(\d+)\s*ms', content)
        if match:
            results.append({
                'name': f'{label} p99 Latency',
                'unit': 'ms',
                'value': int(match.group(1))
            })

    print(json.dumps(results, indent=2))

if __name__ == '__main__':
    parse_benchmark(sys.argv[1])
```
