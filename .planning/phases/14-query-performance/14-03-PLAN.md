   ```zig
       const start_parse = std.time.nanoTimestamp();
       // Parse filter from input...
       const end_parse = std.time.nanoTimestamp();

       const start_plan = std.time.nanoTimestamp();
       // Compute S2 covering (or get from cache)...
       const end_plan = std.time.nanoTimestamp();

       const start_execute = std.time.nanoTimestamp();
       // Scan index and collect results...
       const end_execute = std.time.nanoTimestamp();

       const start_serialize = std.time.nanoTimestamp();
       // Write results to output buffer...
       const end_serialize = std.time.nanoTimestamp();

       // Record breakdown
       self.latency_breakdown.recordPhases(.{
           .query_type = .radius,
           .parse_ns = @intCast(end_parse - start_parse),
           .plan_ns = @intCast(end_plan - start_plan),
           .execute_ns = @intCast(end_execute - start_execute),
           .serialize_ns = @intCast(end_serialize - start_serialize),
       });
   }
   ```

   - commit_query_latest: parse, (no plan), execute, serialize
