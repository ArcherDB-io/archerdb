Create `src/prepared_queries.zig` implementing prepared query storage and compilation:

1. Define CompiledQuery struct (pre-parsed query representation):
   ```zig
   pub const CompiledQuery = struct {
       query_type: QueryType,
       // Pre-validated, normalized filter template
       filter_template: FilterTemplate,
       // Parameter slots for substitution
       param_slots: [max_params]ParamSlot,
       num_params: u8,

       pub const max_params = 8;
   };

   pub const QueryType = enum(u8) {
       uuid = 0,
       radius = 1,
       polygon = 2,
       latest = 3,
   };

   pub const ParamSlot = struct {
       param_type: ParamType,
       offset: u16,  // Offset into filter template
       size: u8,     // Size of parameter
   };

   pub const ParamType = enum(u8) {
       entity_id,      // u128
       lat_nano,       // i64
       lon_nano,       // i64
       radius_mm,      // u32
       timestamp_min,  // u64
       timestamp_max,  // u64
       group_id,       // u64
       limit,          // u32
   };
   ```

2. Define FilterTemplate union for each query type:
   ```zig
   pub const FilterTemplate = union(QueryType) {
       uuid: QueryUuidFilter,
       radius: QueryRadiusFilter,
       polygon: PolygonFilterTemplate,
       latest: QueryLatestFilter,
   };

   pub const PolygonFilterTemplate = struct {
       vertex_count: u32,
       // Polygon vertices stored separately
   };
   ```

3. Define PreparedQuery struct:
   ```zig
   pub const PreparedQuery = struct {
       /// Hash of user-provided name for lookup
       name_hash: u64,
       /// Compiled query representation
       compiled: CompiledQuery,
       /// Execution statistics
       execution_count: u64,
       total_duration_ns: u64,

       pub fn averageExecutionNs(self: PreparedQuery) u64 {
           if (self.execution_count == 0) return 0;
           return self.total_duration_ns / self.execution_count;
       }
   };
   ```

4. Define SessionPreparedQueries (per-session storage):
   ```zig
   pub const SessionPreparedQueries = struct {
       queries: [max_prepared_per_session]?PreparedQuery,
       count: u32,

       pub const max_prepared_per_session = 32;

       pub fn init() SessionPreparedQueries { ... }
       pub fn prepare(self: *SessionPreparedQueries, name: []const u8, query_text: []const u8) !u32 { ... }
       pub fn execute(self: *SessionPreparedQueries, name_or_slot: u32, params: []const u8, output: []u8) !usize { ... }
       pub fn deallocate(self: *SessionPreparedQueries, name_hash: u64) bool { ... }
       pub fn clear(self: *SessionPreparedQueries) void { ... }
   };
   ```

5. Implement query compilation (prepare):
   - Parse query text to determine query type
   - Extract parameter placeholders ($1, $2, etc.)
   - Validate parameter types match query requirements
