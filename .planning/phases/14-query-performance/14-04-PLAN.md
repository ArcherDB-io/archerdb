1. Add executeBatch function:
   ```zig
   pub fn executeBatch(
       state_machine: *GeoStateMachineType,
       input: []const u8,
       output: []u8,
   ) usize {
       // Parse request header
       const request = @as(*const BatchQueryRequest, @ptrCast(@alignCast(input.ptr)));
       var offset: usize = @sizeOf(BatchQueryRequest);

       // Prepare response header
       var response = BatchQueryResponse{
           .total_count = request.query_count,
           .success_count = 0,
           .error_count = 0,
           .has_more = 0,
       };

       // Reserve space for response header and result entries
       var output_offset: usize = @sizeOf(BatchQueryResponse);
       const entries_offset = output_offset;
       output_offset += request.query_count * @sizeOf(BatchQueryResultEntry);
       const results_start = output_offset;

       // Execute each query
       for (0..request.query_count) |i| {
           const entry = @as(*const BatchQueryEntry, @ptrCast(@alignCast(input.ptr + offset)));
           offset += @sizeOf(BatchQueryEntry);

           // Dispatch to appropriate query handler
           const result = executeQuery(state_machine, entry, input[offset..], output[output_offset..]);

           // Record result entry
           const result_entry = BatchQueryResultEntry{
               .query_id = entry.query_id,
               .status = result.status,
               .result_offset = @intCast(output_offset - results_start),
               .result_length = result.length,
           };
           @memcpy(output[entries_offset + i * @sizeOf(BatchQueryResultEntry)..], ...);

           if (result.status == 0) {
               response.success_count += 1;
           } else {
               response.error_count += 1;
           }

           output_offset += result.length;
           offset += filterSizeForType(entry.query_type);

           // Check response size limit
           if (output_offset > constants.message_body_size_max - 1024) {
               response.has_more = 1;
               break;
           }
       }

       // Write response header
       @memcpy(output[0..@sizeOf(BatchQueryResponse)], ...);
       return output_offset;
   }
   ```

       entry: *const BatchQueryEntry,
