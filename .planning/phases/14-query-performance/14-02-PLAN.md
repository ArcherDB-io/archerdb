3. Implement hash functions for cache keys (CRITICAL: use only integers):

   For radius queries:
   ```zig
   pub fn hashCapParams(center_lat_nano: i64, center_lon_nano: i64, radius_mm: u32) u64 {
       // Use stdx.hash_inline for deterministic hashing
       var h: u64 = 0;
       h ^= stdx.hash_inline(@as([8]u8, @bitCast(center_lat_nano)));
       h ^= stdx.hash_inline(@as([8]u8, @bitCast(center_lon_nano))) *% 31;
       h ^= stdx.hash_inline(@as([4]u8, @bitCast(radius_mm))) *% 17;
       return h;
   }
   ```

   For polygon queries:
   ```zig
   pub fn hashPolygonParams(vertices: []const LatLon) u64 {
       // Hash all vertices in order
       var h: u64 = 0;
       for (vertices) |v| {
           h ^= stdx.hash_inline(@as([8]u8, @bitCast(v.lat_nano)));
           h ^= stdx.hash_inline(@as([8]u8, @bitCast(v.lon_nano))) *% 31;
       }
       return h;
   }
   ```

4. Implement cache operations:
   - getCapCovering(lat_nano, lon_nano, radius_mm) -> ?CachedCovering
   - putCapCovering(lat_nano, lon_nano, radius_mm, covering) -> void
   - getPolygonCovering(vertices_hash) -> ?CachedCovering
   - putPolygonCovering(vertices_hash, covering) -> void

5. Add init/deinit following SetAssociativeCache pattern

2. Add cached covering lookup in commit_query_radius:
   ```zig
   // Try cache first
   const covering = if (self.covering_cache.getCapCovering(
       filter.center_lat_nano,
       filter.center_lon_nano,
       filter.radius_mm,
   )) |cached|
       cached.ranges[0..cached.num_ranges]
   else blk: {
       // Compute covering
       const computed = S2.coverCap(scratch, lat, lon, radius_mm, 8, 30);
       // Cache it
       self.covering_cache.putCapCovering(lat, lon, radius_mm, computed);
       break :blk computed;
   };
   ```

3. Add cached covering lookup in commit_query_polygon:

5. Note: S2 covering cache does NOT need write-invalidation

   - Same region always produces same covering
