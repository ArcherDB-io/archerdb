  <name>Task 2: Create query performance alerting rules</name>
  <files>alerts/query-performance.yaml</files>
  <action>
Create `alerts/query-performance.yaml` with alerting rules:

```yaml
groups:
  - name: query_performance
    rules:
      # Cache hit ratio alerts
      - alert: QueryCacheHitRatioLow
        expr: |
          (sum(rate(archerdb_query_cache_hits_total[5m])) /
           (sum(rate(archerdb_query_cache_hits_total[5m])) +
            sum(rate(archerdb_query_cache_misses_total[5m])))) < 0.6
        for: 10m
        labels:
          severity: warning
        annotations:
          summary: "Query cache hit ratio below 60%"
          description: "Cache hit ratio {{ $value | humanizePercentage }} is below target. Check for query pattern changes or cache size."

      - alert: QueryCacheHitRatioCritical
        expr: |
          (sum(rate(archerdb_query_cache_hits_total[5m])) /
           (sum(rate(archerdb_query_cache_hits_total[5m])) +
            sum(rate(archerdb_query_cache_misses_total[5m])))) < 0.4
        for: 5m
        labels:
          severity: critical
        annotations:
          summary: "Query cache hit ratio critically low"
          description: "Cache hit ratio {{ $value | humanizePercentage }} is critically low. Query performance degraded."

      # Query latency alerts
      - alert: QueryLatencyHigh
        expr: histogram_quantile(0.99, sum(rate(archerdb_query_total_seconds_bucket[5m])) by (le)) > 0.1
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "Query P99 latency exceeds 100ms"
          description: "P99 query latency is {{ $value | humanizeDuration }}. Check cache effectiveness and index health."

      - alert: QueryLatencyCritical
        expr: histogram_quantile(0.99, sum(rate(archerdb_query_total_seconds_bucket[5m])) by (le)) > 0.5
        for: 2m
        labels:
          severity: critical
        annotations:
          summary: "Query P99 latency exceeds 500ms"
          description: "P99 query latency is {{ $value | humanizeDuration }}. Immediate investigation required."

      # Covering cache alerts
      - alert: S2CoveringCacheHitRatioLow
        expr: |
          (sum(rate(archerdb_s2_covering_cache_hits_total[5m])) /
           (sum(rate(archerdb_s2_covering_cache_hits_total[5m])) +
            sum(rate(archerdb_s2_covering_cache_misses_total[5m])))) < 0.5
        for: 15m
        labels:
          severity: info
        annotations:
          summary: "S2 covering cache hit ratio below 50%"
          description: "Spatial queries may be using diverse geographic regions. Consider cache size increase."

      # Prepared query alerts
      - alert: PreparedQueryCompileErrorsHigh
        expr: sum(rate(archerdb_prepared_query_errors_total[5m])) > 1
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "Prepared query compile errors elevated"
          description: "{{ $value }} prepared query errors per second. Check application query syntax."

      # Batch query alerts
      - alert: BatchQuerySuccessRateLow
        expr: avg(archerdb_batch_query_success_rate) < 0.9
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "Batch query success rate below 90%"
          description: "Batch queries have {{ $value | humanizePercentage }} success rate. Check individual query errors."

      # RAM index alerts (from spatial stats)
      - alert: RAMIndexLoadFactorHigh
        expr: archerdb_ram_index_load_factor / 1000 > 0.7
        for: 10m
        labels:
          severity: warning
        annotations:
          summary: "RAM index load factor above 70%"
          description: "RAM index at {{ $value | humanizePercentage }} load. Consider capacity increase."
```

Follow existing alert patterns from alerts/archerdb-alerts.yaml for consistency.
